68  JsonValueStoreDeprecatedReader(std::istream& stream,
69  boost::interprocess::file_mapping* file_mapping,
70  bool load_lazy = false)
71  : IValueStoreReader(stream, file_mapping) {
72 
73  TRACE("JsonValueStoreDeprecatedReader construct");
74 
75  properties_ =
76  internal::SerializationUtils::ReadJsonRecord(stream);
77 
78  size_t offset = stream.tellg();
79  size_t strings_size = boost::lexical_cast<size_t>(properties_.get<std::string>("size"));
80 
81  // check for file truncation
82  if (strings_size > 0) {
83  stream.seekg(strings_size - 1, stream.cur);
84  if (stream.peek() == EOF) {
85  throw std::invalid_argument("file is corrupt(truncated)");
86  }
87  }
88 
89  boost::interprocess::map_options_t map_options = boost::interprocess::default_map_options;
90 
91 #ifdef MAP_HUGETLB
92  map_options |= MAP_HUGETLB;
93 #endif
94 
95  if (!load_lazy) {
96 #ifdef MAP_POPULATE
97  map_options |= MAP_POPULATE;
98 #endif
99  }
100 
101  strings_region_ = new boost::interprocess::mapped_region(
102  *file_mapping, boost::interprocess::read_only, offset,
103  strings_size, 0, map_options);
104 
105  strings_ = (const char*) strings_region_->get_address();
106  }

136  if (packed_string[0] == ' ') {

167  std::string decompress_string(const std::string& str) const
168  {
169  z_stream zs; // z_stream is zlib's control structure
170  memset(&zs, 0, sizeof(zs));
171 
172  if (inflateInit(&zs) != Z_OK)
173  throw(std::runtime_error("inflateInit failed while decompressing."));
174 
175  zs.next_in = (Bytef*)str.data() + 1;
176  zs.avail_in = str.size() - 1;
177 
178  int ret;
179  char outbuffer[32768];
180  std::string outstring;
181 
182  // get the decompressed bytes blockwise using repeated calls to inflate
183  do {
184  zs.next_out = reinterpret_cast<Bytef*>(outbuffer);
185  zs.avail_out = sizeof(outbuffer);
186 
187  ret = inflate(&zs, 0);
188 
189  if (outstring.size() < zs.total_out) {
190  outstring.append(outbuffer,
191  zs.total_out - outstring.size());
192  }
193 
194  } while (ret == Z_OK);
195 
196  inflateEnd(&zs);
197 
198  if (ret != Z_STREAM_END) { // an error occurred that was not EOF
199  std::ostringstream oss;
200  oss << "Exception during zlib decompression: (" << ret << ") "
201  << zs.msg;
202  throw(std::runtime_error(oss.str()));
203  }
204 
205  return outstring;
206  }
