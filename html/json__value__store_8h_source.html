76  JsonValueStore(const vs_param_t& parameters = vs_param_t(), size_t memory_limit = 104857600)

80  "dictionary-fsa-json_value_store-%%%%-%%%%-%%%%-%%%%");
81  boost::filesystem::create_directory(temporary_directory_);
82 
83  if (parameters_.count(COMPRESSION_THRESHOLD_KEY) > 0) {
84  compression_threshold_ = boost::lexical_cast<size_t>(
85  parameters_[COMPRESSION_THRESHOLD_KEY]);
86  } else {
87  compression_threshold_ = 32;
88  }
89 
90  std::string compressor;
91  if (parameters_.count(COMPRESSION_KEY) > 0) {
92  compressor = parameters_[COMPRESSION_KEY];
93  }
94 
95  if (parameters_.count(MINIMIZATION_KEY) > 0 && parameters_[MINIMIZATION_KEY] == "off") {
96  minimize_ = false;
97  }
98 
99  if (parameters_.count(SINGLE_PRECISION_FLOAT_KEY) > 0 && parameters_[SINGLE_PRECISION_FLOAT_KEY] == "single") {
100  // set single precision float mode
101  msgpack_buffer_.set_single_precision_float();
102  }
103 
104  compressor_.reset(compression::compression_strategy(compressor));
105  raw_compressor_.reset(compression::compression_strategy("raw"));
106  // This is beyond ugly, but needed for EncodeJsonValue :(
107  long_compress_ = std::bind(static_cast<compression::compress_mem_fn_t>
108  (&compression::CompressionStrategy::Compress),
109  compressor_.get(), std::placeholders::_1, std::placeholders::_2, std::placeholders::_3);
110  short_compress_ = std::bind(static_cast<compression::compress_mem_fn_t>
111  (&compression::CompressionStrategy::Compress),
112  raw_compressor_.get(), std::placeholders::_1, std::placeholders::_2, std::placeholders::_3);
113 
114  size_t external_memory_chunk_size = 1073741824;
115 
116  values_extern_.reset(new MemoryMapManager(external_memory_chunk_size, temporary_directory_,
117  "json_values_filebuffer"));

153  uint64_t GetValue(const value_t& value, bool& no_minimization) {
154  msgpack_buffer_.clear();
155 
156  util::EncodeJsonValue(
157  long_compress_, short_compress_, msgpack_buffer_,
158  string_buffer_, value,
159  compression_threshold_);
160 
161  ++number_of_values_;
162 
163  if (!minimize_){
164  TRACE("Minimization is turned off.");
165  no_minimization = true;
166  return AddValue();
167  }
168 
169  const RawPointerForCompare<MemoryMapManager> stp(string_buffer_.data(), string_buffer_.size(), values_extern_.get());
170  const RawPointer<> p = hash_.Get(stp);
171 
172  if (!p.IsEmpty()) {
173  // found the same value again, minimize
174  TRACE("Minimized value");
175  return p.GetOffset();
176  } // else persist string value
177 
178  no_minimization = true;
179  TRACE("New unique value");
180  ++number_of_unique_values_;
181 
182  uint64_t pt = AddValue();
183 
184  TRACE("add value to hash at %d, length %d", pt, string_buffer_.size());
185  hash_.Add(RawPointer<>(pt, stp.GetHashcode(), string_buffer_.size()));
186 
187  return pt;
188  }

211  void Write(std::ostream& stream) {
212  boost::property_tree::ptree pt;
213  pt.put("size", std::to_string(values_buffer_size_));
214  pt.put("values", std::to_string(number_of_values_));
215  pt.put("unique_values", std::to_string(number_of_unique_values_));
216 
217  if (!mergeMode_) {
218  pt.put(std::string("__") + COMPRESSION_KEY, compressor_->name());
219  pt.put(std::string("__") + COMPRESSION_THRESHOLD_KEY, compression_threshold_);
220  }
221 
222  internal::SerializationUtils::WriteJsonRecord(stream, pt);
223  TRACE("Wrote JSON header, stream at %d", stream.tellp());
224 
225  if (!mergeMode_) {
226  values_extern_->Write(stream, values_buffer_size_);
227  } else {
228  for (const auto& filename: inputFiles_) {
229  KeyViFile keyViFile(filename);
230  auto &in_stream = keyViFile.valueStoreStream();
231  internal::SerializationUtils::ReadValueStoreProperties(in_stream);
232 
233  stream << in_stream.rdbuf();
234  }
235  }

269  uint64_t GetValue(const char* payload, uint64_t fsa_value, bool& no_minimization){
270  size_t buffer_size;
271 
272  const char* full_buf = payload + fsa_value;
273  const char* buf_ptr = util::decodeVarintString(full_buf, &buffer_size);
274 
275  const RawPointerForCompare<MemoryMapManager> stp(buf_ptr, buffer_size, values_extern_.get());
276  const RawPointer<> p = hash_.Get(stp);
277 
278  if (!p.IsEmpty()) {
279  // found the same value again, minimize
280  TRACE("Minimized value");
281  return p.GetOffset();
282  } // else persist string value
283 
284  no_minimization = true;
285  TRACE("New unique value");
286  ++number_of_unique_values_;
287 
288  uint64_t pt = static_cast<uint64_t>(values_buffer_size_);
289  size_t full_buf_size = (buf_ptr-full_buf) + buffer_size;
290 
291  values_extern_->Append((void*)full_buf,
292  full_buf_size);
293  values_buffer_size_ += full_buf_size;
294 
295  hash_.Add(RawPointer<>(pt, stp.GetHashcode(), buffer_size));
296 
297  return pt;
298  }

300  uint64_t AddValue(){
301  uint64_t pt = static_cast<uint64_t>(values_buffer_size_);
302  size_t length;
303 
304  dictionary::util::encodeVarint(string_buffer_.size(), *values_extern_, &length);
305  values_buffer_size_ += length;
306  values_extern_->Append((void*)string_buffer_.data(),
307  string_buffer_.size());
308  values_buffer_size_ += string_buffer_.size();
309 
310  return pt;
