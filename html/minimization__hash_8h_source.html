92  MemoryLimitConfiguration FindMemoryLimitConfiguration (
93  size_t memory_limit, int min, int max) const {
94  int best_fit_memory_limit = 0;
95  int best_fit_generations = 0;
96  int best_fit_maximum_number_of_items_per_table = 0;
97 
98  // try to find a good number of generations (between given minimum and maximum) with equal size given the memory Limit
99  for (int i = min; i <= max; ++i) {
100  int maximum_number_of_items = 0;
101 
102  // find the value that fits to it in the HashmapSteptable
103  for (unsigned int step = 3; step < kHashMaxSizeStep; ++step) {
104  // the memory usage is the size of the hashtable itself and the size of the array for the overflow buckets which is the number of buckets divided by 4.
105  size_t item_forecast_for_hashtable = kHashSizeStepTable[step]
106  + std::min(kHashSizeStepTable[step] >> 2, max_cookie_size_);
107 
108  size_t memoryUsageForeCast = item_forecast_for_hashtable * sizeof(T)
109  * i;
110 
111  if (memory_limit < memoryUsageForeCast) {
112  maximum_number_of_items = kHashSizeStepTable[step - 1];
113  break;
114  }
115  }
116 
117  // calculate memory usage
118  int usage = (maximum_number_of_items + (maximum_number_of_items >> 2))
119  * sizeof(T) * i;
120  if (usage > best_fit_memory_limit) {
121  best_fit_memory_limit = usage;
122  best_fit_generations = i;
123  best_fit_maximum_number_of_items_per_table =
124  (int) (maximum_number_of_items * kLoadFactor);
125  }
126  }
127 
128  if (best_fit_maximum_number_of_items_per_table == 0) {
129  // todo: exception?
130  }
131 
132  return MemoryLimitConfiguration(
133  best_fit_memory_limit, best_fit_generations,
134  best_fit_maximum_number_of_items_per_table);
135  }

176  template<typename EqualityType>
177  inline const T Get(EqualityType& key) const {
178  int hash = key.GetHashcode() & 0x7fffffff;
179  int bucket = hash % hash_size_;
180 
181  T entry = entries_[bucket];
182  while (!entry.IsEmpty()) {
183  if (key == entry) {
184  return (entry);
185  }
186 
187  int overflowBucket = entry.GetCookie();
188  if (overflowBucket != 0) {
189  entry = overflow_entries_[overflowBucket];

191  return T();

193  }

205  template<typename EqualityType>
206  inline const T GetAndMove(EqualityType& key, MinimizationHash<T>* other) {
207  int hash = key.GetHashcode() & 0x7fffffff;
208  int bucket = hash % hash_size_;
209  T entry = entries_[bucket];
210 
211  if (!entry.IsEmpty()) {
212  int overflow_bucket;
213  if (key == entry) {
214  // delete the old entry
215  overflow_bucket = entry.GetCookie();
216  if (overflow_bucket != 0) {
217  // overwrite with the entry from overflow
218  entries_[bucket] = overflow_entries_[overflow_bucket];
219  }
220 
221  entry.SetCookie(0);
222  other->Add(entry);
223  return entry;
224  }
225 
226  // check for more items in overflow
227  overflow_bucket = entry.GetCookie();
228  if (overflow_bucket != 0) {
229  entry = overflow_entries_[overflow_bucket];
230  if (key == entry) {
231  // disconnect this entry
232  entries_[bucket].SetCookie(entry.GetCookie());
233 
234  entry.SetCookie(0);
235  other->Add(entry);
236  return entry;
237  }
238 
239  // search further in overflowEntries
240  overflow_bucket = entry.GetCookie();
241  entry = overflow_entries_[overflow_bucket];
242 
243  while (!entry.IsEmpty()) {
244  if (key == entry) {
245  // disconnect entry
246  overflow_entries_[overflow_bucket].SetCookie(entry.GetCookie());
247 
248  entry.SetCookie(0);
249  other->Add(entry);
250 
251  return entry;
252  }
253 
254  overflow_bucket = entry.GetCookie();
255  entry = overflow_entries_[overflow_bucket];
256  }
257  }
258  }
259 
260  return T();
261  }

349  inline void Insert(const T key) {
350  int hash = key.GetHashcode() & 0x7fffffff;
351  int bucket = hash % hash_size_;
352 
353  if (entries_[bucket].IsEmpty()) {
354  entries_[bucket] = key;
355  } else {
356  // overflow handling
357 
358  // overflowing is limited to the address space in PrivateUse, if we run out we drop the entries
359  if (overflow_count_ == max_cookie_size_) {
360  return;
361  }
362 
363  int overflowBucket = entries_[bucket].GetCookie();
364 
365  if (overflowBucket == 0) {
366  entries_[bucket].SetCookie(overflow_count_);
367  overflow_entries_[overflow_count_++] = key;
368  } else {
369  size_t number_of_overflows = 0;
370  while (overflow_entries_[overflowBucket].GetCookie() != 0 && number_of_overflows < overflow_limit_) {
371  overflowBucket = overflow_entries_[overflowBucket].GetCookie();
372  ++number_of_overflows;
373  }
374 
375  if (number_of_overflows == overflow_limit_) {
376  TRACE("Drop entry to avoid overflow");
377  return;
378  }
379 
380  overflow_entries_[overflowBucket].SetCookie(overflow_count_);
381  overflow_entries_[overflow_count_++] = key;
382  }
383  }
384  }

389  inline void GrowAndRehash() {
390  hash_size_step_++;
391 
392  TRACE("Re-size hash table(type: %s): size: %zd, overflow size: %zd entries:%zd, collisions: %zd",
393  typeid(T).name(), hash_size_, overflow_entries_size_, count_, overflow_count_);
394 
395  int oldHashSize = hash_size_;
396 
397  hash_size_ = kHashSizeStepTable[hash_size_step_];
398  rehash_limit_ = static_cast<int>(hash_size_ * kLoadFactor);
399 
400  T* old_entries = entries_;
401  entries_ = new T[hash_size_];
402 
403  T* old_overflow_entries = overflow_entries_;
404  overflow_entries_size_ = std::min(hash_size_ >> 2, max_cookie_size_);
405  overflow_entries_ = new T[overflow_entries_size_];
406 
407  int old_overflow_count = overflow_count_;
408  overflow_count_ = 1;
409 
410  for (int i = 0; i < oldHashSize; ++i) {
411  T e = old_entries[i];
412  if (!e.IsEmpty()) {
413  // clear PrivateUse
414  e.SetCookie(0);
415  Insert(e);
416  }
417  }
418 
419  // overflowEntries[0] does not exist, therefore starting with 1
420  for (int i = 1; i < old_overflow_count; ++i) {
421  T e = old_overflow_entries[i];
422 
423  // clear PrivateUse
424  e.SetCookie(0);
425  Insert(e);
426  }
427 
428  delete[] old_entries;
429  delete[] old_overflow_entries;
430  }
431  };
