77  bool Contains(const std::string& key) const {
78  uint64_t state = fsa_->GetStartState();
79  const size_t key_length = key.size();
80 
81  TRACE("Contains for %s", key.c_str());
82  for (size_t i = 0; i < key_length; ++i) {
83  state = fsa_->TryWalkTransition(state, key[i]);
84 
85  if (!state) {
86  return false;
87  }
88  TRACE ("Contains matched %d/%d", i+1, key_length);
89  }
90 
91  TRACE("Contains matched key, looking for Final State (%d)", state);
92  if (state && fsa_->IsFinalState(state)) {
93  TRACE("Contains matched final state.");
94  return true;
95  }
96 
97  return false;
98  }

317  iterators.push(Lookup(text, position).begin());

321  iterators.pop();

324  while (iterators.size() && (*current_it).IsEmpty()) {

347  MatchIterator::MatchIteratorPair GetNear(const std::string& key, size_t minimum_prefix_length, bool greedy=false) {
348  uint64_t state = fsa_->GetStartState();
349 
350  if (key.size() < minimum_prefix_length) {
351  return MatchIterator::EmptyIteratorPair();
352  }
353 
354  TRACE("GetNear %s, matching prefix first", key.substr(0, minimum_prefix_length).c_str());
355  for (size_t i = 0; i < minimum_prefix_length; ++i) {
356  state = fsa_->TryWalkTransition(state, key[i]);
357 
358  if (!state) {
359  return MatchIterator::EmptyIteratorPair();
360  }
361  }
362 
363  std::vector<unsigned char> traversal_stack;
364  traversal_stack.reserve(20);
365 
366  TRACE("Match prefix, doing near traversal now start state: %ld, remaining key: %s", state, key.substr(minimum_prefix_length).c_str());
367

371  std::vector<unsigned char>& stack)

373  traversal_stack(std::move(stack)) {
374  }
375 
376  fsa::NearStateTraverser traverser;
377  std::vector<unsigned char> traversal_stack;
378  size_t matched_depth = 0;
379  };
380

382  std::shared_ptr<delegate_payload> data(
383  new delegate_payload(
384  fsa::NearStateTraverser(fsa_, state, payload),
385  traversal_stack));
386 
387  auto tfunc =
388  [data, key, minimum_prefix_length, greedy] () {
389  TRACE("prefix completion callback called");
390 
391 
392  for (;;) {
393  // check minimum depth
394  if (!data->traverser.AtEnd() && data->traverser.GetDepth() > data->matched_depth) {
395 
396  data->traversal_stack.resize(data->traverser.GetDepth()-1);
397  data->traversal_stack.push_back(data->traverser.GetStateLabel());
398  TRACE("Current depth %d (%d)", minimum_prefix_length + data->traverser.GetDepth() -1, data->traversal_stack.size());
399  if (data->traverser.IsFinalState()) {
400  // optimize? fill vector upfront?
401  std::string match_str = key.substr(0, minimum_prefix_length) + std::string(reinterpret_cast<char*> (&data->traversal_stack[0]), data->traverser.GetDepth());
402  TRACE("found final state at depth %d %s", minimum_prefix_length + data->traverser.GetDepth(), match_str.c_str());
403  Match m(0,
404  data->traverser.GetDepth() + key.size(),
405  match_str,
406  minimum_prefix_length + data->traverser.GetTraversalPayload().exact_depth,
407  data->traverser.GetFsa(),
408  data->traverser.GetStateValue());
409 
410  if (!greedy) {
411  // remember the depth
412  TRACE("found a match, remember depth, only allow matches with same depth %ld", data->traverser.GetTraversalPayload().exact_depth);
413  data->matched_depth = data->traverser.GetTraversalPayload().exact_depth;
414  }
415 
416  data->traverser++;
417 
418  return m;
419  }
420  data->traverser++;
421  } else {
422  TRACE("StateTraverser exhausted.");
423  return Match();
424  }

426  };
