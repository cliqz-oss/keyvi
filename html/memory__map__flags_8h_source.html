42 enum class loading_strategy_types {
43  default_os, // no special treatment, use whatever the OS/Boost has as default
44  lazy, // load data as needed with some read-ahead
45  populate, // immediately load everything in memory (blocks until everything is fully read)
46  populate_key_part, // populate only the key part, load value part lazy
47  populate_lazy, // load data lazy but ask the OS to read ahead if possible (does not block)
48  lazy_no_readahead, // disable any read-ahead (for cases when index > x * main memory)
49  lazy_no_readahead_value_part, // disable read-ahead only for the value part
50  populate_key_part_no_readahead_value_part // populate the key part, but disable read ahead value part
51  };

64  static int FSAGetMemoryMapOptions(const loading_strategy_types strategy) {
65 #ifdef _Win32
66 
67  // there is no comparable fine-grained control on windows, so simply use the defaults
68  return boost::interprocess::default_map_options;
69 #else // not _Win32
70 
71  if (strategy == loading_strategy_types::default_os) {
72  return boost::interprocess::default_map_options;
73  }
74 
75  int flags = 0;
76 
77  flags |= MAP_SHARED;
78 
79 #ifdef MAP_NOSYNC
80  flags |= MAP_NOSYNC
81 #endif
82 
83  switch (strategy){
84  case loading_strategy_types::populate:
85  case loading_strategy_types::populate_key_part:
86  case loading_strategy_types::populate_key_part_no_readahead_value_part:
87  flags |= MAP_POPULATE;
88  break;
89  default:
90  break;
91  }
92 
93  return flags;
94 #endif // not _Win32
95  }
