126  void operator++(int) {
127 
128  // ignore cases where we are already at the end
129  if (at_end_) {
130  TRACE("end of traversal");
131  return;
132  }
133 
134  TRACE("finding next state");
135  traversal_entry_t traversal_entry;
136  for (;;) {
137  uint64_t child_node = 0;
138  uint32_t weight = 0;
139  traversal_entry = entry_traversal_stack_.back();
140  TRACE("remaining transitions at %d %d", current_depth_, traversal_entry.size());
141 
142  if (traversal_entry.size()) {
143  // in case the priority queue got better items between creating the transition list and now,
144  // we have to skip here
145  // note: child_node remains 0, so all remaining transitions are dropped later on
146  weight = traversal_entry.front().first;
147  if (weight >= priority_queue_.Back()) {
148  current_label_ = traversal_entry.front().second;;
149  entry_traversal_stack_.back().pop_front();
150  child_node = fsa_->TryWalkTransition(current_state_, current_label_);
151  TRACE ("Walk transition with label %c (%d)", current_label_, current_label_);
152  }
153  }
154 
155  if (child_node) {
156  /* Found a valid child node
157  * go one level down
158  */
159  TRACE("found child node");
160 
161  //current_label_ = traversal_stack_[current_depth_];
162  ++current_depth_;
163  state_traversal_stack_.push_back(current_state_);
164  current_state_ = child_node;
165 
166  // todo: refactor
167  TRACE("moving down to depth: %d", current_depth_);
168  GetNextTransitionsInSortedOrder(weight);
169 
170  return;
171 
172  } else {
173  // did not found any more transitions at this deep
174 
175  if (current_depth_) {
176  /* we did not find any path at the current level (deep)
177  * go one level up
178  */
179  TRACE("current path exhausted going up");
180 
181  //traversal_stack_.pop_back();
182  current_state_ = state_traversal_stack_.back();
183  state_traversal_stack_.pop_back();
184  entry_traversal_stack_.pop_back();
185  --current_depth_;
186  } else {
187  // we are at the very end
188  TRACE("reached the end, all states traversed");
189 
190  current_state_ = 0;
191  current_depth_ = 0;
192  current_label_ = 0;
193  at_end_ = true;
194  return;
195  }
196  }
197  }

240  void GetNextTransitionsInSortedOrder(uint32_t parent_weight) {
241  uint64_t child_node;
242  traversal_entry_t outgoing_transitions;
243  for (int i = 1; i < 256; ++i) {
244  child_node = fsa_->TryWalkTransition(current_state_, i);
245  if (child_node) {
246  // todo: stop reading the weight dependent on the depth of traversal
247  uint32_t weight = fsa_->GetWeightValue(child_node);
248 
249  // if weight is not set take the weight of the parent
250  if (weight == 0) {
251  TRACE("Received no weight, take weight from parent.");
252  weight = parent_weight;
253  }
254 
255  TRACE("transition with weight %d queue last: %d", weight, priority_queue_.Back());
256  if (weight < priority_queue_.Back()) {
257  continue;
258  } else if (parent_weight != weight && weight > priority_queue_.Back()) {
259  priority_queue_.Put(weight);
260  }
261 
262  outgoing_transitions.push_back(
263  std::pair<uint32_t, unsigned char>(weight, i));
264 
265  }
266  }
267 
268  std::sort(outgoing_transitions.begin(), outgoing_transitions.end(),
269  compare);
270  TRACE("first transition after sort: %c %d", outgoing_transitions[0].second, outgoing_transitions[0].first);
271 
272  TRACE("number of transitions found: %d", outgoing_transitions.size());
273  entry_traversal_stack_.push_back(outgoing_transitions);
274  }
