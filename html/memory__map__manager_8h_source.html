64  /* Using GetAdress to read multiple bytes is unsafe as it might be a buffer overflow
65  *
66  * This API is to check first whether GetAdress is safe to use.
67  */
68  bool GetAddressQuickTestOk(size_t offset, size_t length) const {
69  size_t chunk_offset = offset % chunk_size_;
70 
71  return (length <= (chunk_size_ - chunk_offset));
72  }

74  void* GetAddress(size_t offset) {
75  size_t chunk_number = offset / chunk_size_;
76  size_t chunk_offset = offset % chunk_size_;

83  /* Get a buffer as copy.
84  *
85  * This API is to be used when GetAdress is not safe to use.
86  */
87  void GetBuffer(const size_t offset, void* buffer, const size_t buffer_length) {
88  size_t chunk_number = offset / chunk_size_;
89  size_t chunk_offset = offset % chunk_size_;
90 
91  void* chunk_address = GetChunk(chunk_number);
92  void* chunk_address_part2 = GetChunk(chunk_number + 1);
93 
94  size_t first_chunk_size = std::min(buffer_length, chunk_size_ - chunk_offset);
95  size_t second_chunk_size = buffer_length - first_chunk_size;
96 
97  std::memcpy(buffer, (char*) chunk_address + chunk_offset, first_chunk_size);
98  std::memcpy((char*) buffer + first_chunk_size, (char*) chunk_address_part2, second_chunk_size);
99  }

108  void Append(const void* buffer, const size_t buffer_length){
109  size_t remaining = buffer_length;
110  size_t buffer_offset = 0;
111 
112  TRACE("append %ld %ld", buffer_offset, remaining);
113 
114  while (remaining > 0) {
115  TRACE("next chunk remaining: %ld", remaining);
116 
117  size_t chunk_number = tail_ / chunk_size_;
118  size_t chunk_offset = tail_ % chunk_size_;
119  TRACE ("chunk number: %ld offset %ld", chunk_number, chunk_offset);
120 
121  void* chunk_address = GetChunk(chunk_number);
122  size_t copy_size = std::min(remaining, chunk_size_ - chunk_offset);
123  TRACE ("copy size: %ld", copy_size);
124 
125  std::memcpy((char*)chunk_address + chunk_offset, (char*)buffer + buffer_offset, copy_size);
126 
127  remaining -= copy_size;
128  tail_ += copy_size;
129  buffer_offset += copy_size;
130  }
131  }

133  void push_back(const char c){
134  size_t chunk_number = tail_ / chunk_size_;
135  size_t chunk_offset = tail_ % chunk_size_;
136 
137  char* chunk_address = static_cast<char*> (GetChunk(chunk_number));
138  chunk_address[chunk_offset] = c;
139  ++tail_;
140  }

142  bool Compare(const size_t offset, const void* buffer, const size_t buffer_length){
143  size_t chunk_number = offset / chunk_size_;
144  size_t chunk_offset = offset % chunk_size_;
145 
146  void* chunk_address = GetChunk(chunk_number);
147  size_t first_chunk_size = std::min(buffer_length, chunk_size_ - chunk_offset);
148 
149  if (std::memcmp((char*)chunk_address + chunk_offset, buffer, first_chunk_size) != 0) {
150  return false;
151  }
152 
153  // no overflow and equal
154  if (buffer_length == first_chunk_size) {
155  return true;
156  }
157 
158  // handle overflow
159  void* chunk_address_part2 = GetChunk(chunk_number + 1);
160  return (std::memcmp((char*)chunk_address_part2, (char*)buffer + first_chunk_size, buffer_length - first_chunk_size) == 0);
161  }

163  void Write (std::ostream& stream, const size_t end) const {
164  if (persisted_) {
165  for (size_t i =0; i < number_of_chunks_; i++)
166  {
167  std::ifstream data_file;
168  data_file.open (GetFilenameForChunk(i).native().c_str(), std::ios::binary);
169  stream << data_file.rdbuf();
170  data_file.close();
171  }
172  } else if (number_of_chunks_ == 0) {
173  return;
174  }else {
175  size_t remaining = end;
176  int chunk = 0;
177 
178  while (remaining > 0) {
179  size_t bytes_in_chunk = std::min(chunk_size_, remaining);
180  TRACE("write chunk %d, with size: %ld, remaining: %ld", i, bytes_in_chunk, remaining);
181 
182  char *ptr = (char*) mappings_[chunk].region_->get_address();
183  stream.write (ptr, bytes_in_chunk);
184 
185  remaining -= bytes_in_chunk;
186  ++chunk;
187  }
188  }
189  }

198  void Persist() {
199  persisted_ = true;
200  for (auto& m : mappings_) {
201  m.region_->flush();
202  delete m.region_;
203  delete m.mapping_;
204  }
205 
206  // truncate last file according to the written buffers
207  if (number_of_chunks_ > 0) {
208  boost::filesystem::resize_file(GetFilenameForChunk(number_of_chunks_ - 1), tail_ - ((number_of_chunks_ - 1) * chunk_size_));
209  }
210 
211  mappings_.clear();
212  }

236  void* GetChunk(const size_t chunk_number) {
237  while (chunk_number >= number_of_chunks_) {
238  CreateMapping();
239  }
240 
241  return mappings_[chunk_number].region_->get_address();
242  }

244  void CreateMapping() {
245  TRACE("create new mapping %d", number_of_chunks_ + 1);
246  mapping new_mapping;
247 
248  boost::filesystem::path filename = GetFilenameForChunk(number_of_chunks_);
249 
250  std::filebuf fbuf;
251  fbuf.open(
252  filename.native().c_str(),
253  std::ios_base::in | std::ios_base::out | std::ios_base::trunc
254  | std::ios_base::binary);
255  // Set the size
256  fbuf.pubseekoff(chunk_size_ - 1, std::ios_base::beg);
257  fbuf.sputc(0);
258  fbuf.close();
259 
260  new_mapping.mapping_ = new boost::interprocess::file_mapping(
261  filename.native().c_str(), boost::interprocess::read_write);
262 
263  new_mapping.region_ = new boost::interprocess::mapped_region(
264  *new_mapping.mapping_, boost::interprocess::read_write);
265 
266  // prevent pre-fetching pages by the OS which does not make sense as values usually fit into few pages
267  new_mapping.region_->advise(boost::interprocess::mapped_region::advice_types::advice_random);
268 
269  mappings_.push_back(new_mapping);
270  ++number_of_chunks_;
271  }
