53  static boost::property_tree::ptree ReadJsonRecord(std::istream& stream) {
54  uint32_t header_size;
55  stream.read(reinterpret_cast<char*>(&header_size), sizeof(int));
56  header_size = htonl(header_size);
57  char * buffer = new char[header_size];
58  stream.read(buffer, header_size);
59  std::string buffer_as_string(buffer, header_size);
60  delete[] buffer;
61  std::istringstream string_stream(buffer_as_string);
62 
63  boost::property_tree::ptree properties;
64  boost::property_tree::read_json(string_stream, properties);
65  return properties;
66  }

68  static boost::property_tree::ptree ReadValueStoreProperties(std::istream &stream) {
69  const auto properties = ReadJsonRecord(stream);
70  const auto offset = stream.tellg();
71 
72  // check for file truncation
73  const size_t vsSize = boost::lexical_cast<size_t>(properties.get<std::string>("size"));
74  if (vsSize > 0) {
75  stream.seekg(vsSize - 1, stream.cur);
76  if (stream.peek() == EOF) {
77  throw std::invalid_argument("file is corrupt(truncated)");
78  }
79  }
80 
81  stream.seekg(offset);
82  return properties;
83  }
