83  int Put(int codepoint, int position) {
84  int row = position + 1;
85 
86  // ensure that we have enough rows in the matrix
87  EnsureCapacity(row + 1);
88  compare_sequence_[position] = codepoint;
89 
90  last_put_position_ = position;
91  int columns = distance_matrix_.Columns();
92 
93  // Implementation of so called Ukkonen's cutoff.
94  // The basic idea is to only calculate the cells of the matrix which are
95  // really necessary given the maximum distance. You only need to
96  // calculate the corridor, see the following picture for illustration
97  // (+ calculated, - not calculated ):
98  //
99  // +++++---
100  // -+++++--
101  // --+++++-
102  // ---+++++
103 
104  // 3 because minimum from MaxDistance + 1, IntermediateScore + 1, row + 1
105  int right_cutoff = std::min(
106  columns, row + max_distance_ - intermediate_scores_[row - 1] + 3);
107  int left_cutoff = std::max(1, row - max_distance_);
108  int intermediate_score = INT_MAX;
109  if (left_cutoff > columns) {
110  intermediate_scores_[row] = INT_MAX;
111  return INT_MAX;
112  }
113 
114  // initialize the first cell according to Ukkonen's cutoff
115  distance_matrix_.Set(
116  row,
117  left_cutoff - 1,
118  (row - left_cutoff + 1)
119  * cost_function_.GetInsertionCost());
120 
121  for (int i = left_cutoff; i < right_cutoff; ++i) {
122  int field_result;
123 
124  // 1. check for exact match according to the substitution cost
125  // function
126  int substitution_cost = cost_function_.GetSubstitutionCost(input_sequence_[i - 1], codepoint);
127 
128  int substitution_result = substitution_cost
129  + distance_matrix_.Get(row - 1, i - 1);
130 
131  if (substitution_cost == 0) {
132  // codePoints match
133  field_result = substitution_result;
134  } else {
135  // 2. calculate costs for deletion, insertion and transposition
136  int deletion_result = distance_matrix_.Get(row, i - 1)
137  + cost_function_.GetDeletionCost();
138  int insertion_result = distance_matrix_.Get(row - 1, i)
139  + cost_function_.GetInsertionCost();
140 
141  int transposition_result = INT_MAX;
142 
143  if (row > 1 && i > 1
144  && input_sequence_[i - 1] == compare_sequence_[position - 1]
145  && input_sequence_[i - 2] == compare_sequence_[position]) {
146  transposition_result = distance_matrix_.Get(row - 2, i - 2)
147  + cost_function_.GetTranspositionCost();
148  }
149 
150  // 4. take the minimum cost
151  //field_result = std::min( { deletion_result, insertion_result,
152  // transposition_result, substitution_result });
153 
154  field_result = std::min(deletion_result, insertion_result);
155 
156  field_result = std::min(field_result, transposition_result);
157 
158  field_result = std::min(field_result, substitution_result);
159  }
160 
161  // put cost into matrix
162  distance_matrix_.Set(row, i, field_result);
163  if (field_result < intermediate_score) {
164  intermediate_score = field_result;
165  }
166  }
167 
168  // set cutoff cell to maxDistance + 1 so that the following row does not
169  // get wrong values due to uninitialized values
170  if (right_cutoff < columns) {
171  distance_matrix_.Set(row, right_cutoff, max_distance_ + 1);
172  distance_matrix_.Set(row, columns - 1, max_distance_ + 1);
173  }
174 
175  latest_calculated_row_ = row;
176 
177  if (intermediate_score > max_distance_) {
178  intermediate_score = max_distance_ + 1;
179  distance_matrix_.Set(row, std::min(row, columns - 1),
180  intermediate_score);
181  distance_matrix_.Set(row, columns - 1, max_distance_ + 1);
182  }
183 
184  intermediate_scores_[row] = intermediate_score;
185 
186  return intermediate_score;
187  }

189  int GetScore() const {
190  return distance_matrix_.Get(latest_calculated_row_,
191  distance_matrix_.Columns() - 1);

194  std::string GetCandidate() {
195  std::vector<unsigned char> utf8result;
196  try {
197  utf8::utf32to8(compare_sequence_,
198  compare_sequence_ + last_put_position_ + 1,
199  back_inserter(utf8result));
200  } catch (utf8::invalid_code_point& e) {
201  //std::cout << "invalid codepoint " << e.code_point() << '\n';
202  return "utf8 support not implemented yet";
203  }
204  return std::string(utf8result.begin(), utf8result.end());
205  }

218  void init(size_t rows) {
219  // initialize first row (sand box row)
220  for (size_t i = 0; i < distance_matrix_.Columns(); ++i) {
221  distance_matrix_.Set(
222  0, i, i * cost_function_.GetDeletionCost());
223  }
224 
225  latest_calculated_row_ = 1;
226 
227  // initialize compare Sequence and immediateScore
228  compare_sequence_ = new int[rows];
229  intermediate_scores_ = new int[rows];
230  intermediate_scores_[0] = 0;
231  compare_sequence_size_ = rows;
232  }

234  void EnsureCapacity(size_t capacity) {
235  // ensure that we have enough rows in the matrix
236  distance_matrix_.EnsureRowCapacity(capacity + 1);
237  if (capacity /*distance_matrix_.Rows()*/> compare_sequence_size_) {
238  // todo: increase by more than 1
239 
240  TRACE("expanding buffers from %d to %d", compare_sequence_size_, capacity);
241  int* compare_sequence_new = new int[capacity];
242  std::memcpy(compare_sequence_new, compare_sequence_,
243  sizeof(int) * compare_sequence_size_);
244 
245  int* intermediate_scores_new = new int[capacity];
246  std::memcpy(intermediate_scores_new, intermediate_scores_,
247  sizeof(int) * compare_sequence_size_);
248 
249  delete[] compare_sequence_;
250  delete[] intermediate_scores_;
251 
252  intermediate_scores_ = intermediate_scores_new;
253  compare_sequence_ = compare_sequence_new;
254  compare_sequence_size_ = capacity;
255  }
256  }
