89  void Add(EntryT key) {
90  if (current_generation_->Size() >= size_per_generation_) {
91  MinimizationHash<EntryT>* newGeneration = NULL;
92  if (generations_.size() + 1 == max_number_of_generations_) {
93  // remove(free) the first generation
94  newGeneration = generations_[0];
95  newGeneration->Reset();
96  generations_.erase(generations_.begin());
97  }
98 
99  generations_.push_back(current_generation_);
100 
101  if (newGeneration == NULL) {
102  newGeneration = new MinimizationHash<EntryT>();
103  }
104 
105  current_generation_ = newGeneration;
106  }
107 
108  current_generation_->Add(key);
109  }

111  template<typename EqualityType>
112  const EntryT Get(EqualityType& key) {
113 
114  EntryT state = current_generation_->Get(key);
115 
116  if (!state.IsEmpty()) {
117  return state;
118  }
119 
120  // try to find it in one of the generations
121  for (size_t i = generations_.size(); i > 0; --i) {
122  state = generations_[i - 1]->GetAndMove(key, current_generation_);
123 
124  if (!state.IsEmpty()) {
125  return state;
126  }
127  }
128 
129  // return am empty state
130  return EntryT();
131  }
