44  std::stringstream string_buffer;
45 
46  boost::property_tree::write_json(string_buffer, properties, false);

49  uint32_t size = ntohl(header.size());

53  }

55  static boost::property_tree::ptree ReadJsonRecord(std::istream& stream) {
56  uint32_t header_size;
57  stream.read(reinterpret_cast<char*>(&header_size), sizeof(int));
58  header_size = htonl(header_size);
59  char * buffer = new char[header_size];
60  stream.read(buffer, header_size);
61  std::string buffer_as_string(buffer, header_size);
62  delete[] buffer;
63  std::istringstream string_stream(buffer_as_string);
64 
65  boost::property_tree::ptree properties;
66  boost::property_tree::read_json(string_stream, properties);
67  return properties;
68  }

70  static boost::property_tree::ptree ReadValueStoreProperties(std::istream &stream) {
71  const auto properties = ReadJsonRecord(stream);
72  const auto offset = stream.tellg();
73 
74  // check for file truncation
75  const size_t vsSize = boost::lexical_cast<size_t>(properties.get<std::string>("size"));
76  if (vsSize > 0) {
77  stream.seekg(vsSize - 1, stream.cur);
78  if (stream.peek() == EOF) {
79  throw std::invalid_argument("file is corrupt(truncated)");
80  }
81  }
82 
83  stream.seekg(offset);
84  return properties;
85  }
