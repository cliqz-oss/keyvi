175  void Compile(callback_t progress_callback = nullptr,
176  void* user_data = nullptr) {
177  size_t added_key_values = 0;
178  size_t callback_trigger = 0;
179 
180  value_store_->CloseFeeding();
181  sorter_.sort();
182  generator_ =
183  GeneratorAdapter::CreateGenerator(size_of_keys_, params_, value_store_);
184  generator_->SetManifest(manifest_);
185 
186  if (sorter_.size() > 0) {
187  size_t number_of_items = sorter_.size();
188 
189  callback_trigger = 1 + (number_of_items - 1) / 100;
190 
191  if (callback_trigger > 100000) {
192  callback_trigger = 100000;
193  }
194 
195  if (!stable_insert_) {
196  for (auto key_value : sorter_) {
197  TRACE("adding to generator: %s", key_value.key.c_str());
198 
199  generator_->Add(std::move(key_value.key), key_value.value);
200  ++added_key_values;
201  if (progress_callback && (added_key_values % callback_trigger == 0)) {
202  progress_callback(added_key_values, number_of_items, user_data);
203  }
204  }
205 
206  } else {
207  // special mode for stable (incremental) inserts, in this case we have
208  // to respect the order and take
209  // the last value if keys are equal
210 
211  auto key_values_it = sorter_.begin();
212  key_value_t last_key_value = *key_values_it++;
213 
214  while (key_values_it != sorter_.end()) {
215  key_value_t key_value = *key_values_it++;
216 
217  // dedup with last one wins
218  if (last_key_value.key == key_value.key) {
219  TRACE("Detected duplicated keys, dedup them, last one wins.");
220 
221  // check the counter to determine which key_value has been added
222  // last
223  if (last_key_value.value.count < key_value.value.count) {
224  last_key_value = key_value;
225  }
226  continue;
227  }
228 
229  if (!last_key_value.value.deleted) {
230  TRACE("adding to generator: %s", last_key_value.key.c_str());
231  generator_->Add(std::move(last_key_value.key),
232  last_key_value.value);
233  ++added_key_values;
234  if (progress_callback &&
235  (added_key_values % callback_trigger == 0)) {
236  progress_callback(added_key_values, number_of_items, user_data);
237  }
238  } else {
239  TRACE("skipping deleted key: %s", last_key_value.key.c_str());
240  }
241 
242  last_key_value = key_value;
243  }
244 
245  // add the last one
246  TRACE("adding to generator: %s", last_key_value.key.c_str());
247  if (!last_key_value.value.deleted) {
248  generator_->Add(std::move(last_key_value.key), last_key_value.value);
249  }
250 
251  ++added_key_values;
252  }
253  }
254 
255  sorter_.clear();
256  generator_->CloseFeeding();
257  }
