95  template<class PersistenceT, class HashCodeTypeT = int32_t>
96  struct RawPointerForCompare final {
97  public:
98  RawPointerForCompare(const char* value, size_t value_size, PersistenceT* persistence)
99  : value_(value), value_size_(value_size), persistence_(persistence) {
100 
101  // calculate a hashcode
102  HashCodeTypeT h = 31;
103 
104  for(size_t i = 0; i < value_size_; ++i) {
105  h = (h * 54059) ^ (value[i] * 76963);
106  }
107 
108  TRACE("hashcode %d", h);
109 
110  hashcode_ = h;
111  }
112 
113  HashCodeTypeT GetHashcode() const {
114  return hashcode_;
115  }
116 
117  bool operator==(const RawPointer<HashCodeTypeT>& l) const {
118  TRACE("check equality, 1st hashcode");
119 
120  // First filter - check if hash code is the same
121  if (l.GetHashcode() != hashcode_) {
122  return false;
123  }
124 
125  TRACE("check equality, 2nd length");
126  size_t length_l = l.GetLength();
127 
128  if (length_l < USHRT_MAX) {
129  if (length_l != value_size_) {
130  return false;
131  }
132 
133  TRACE("check equality, 3rd buffer %d %d %d", l.GetOffset(), value_size_, util::getVarintLength(length_l));
134  // we know the length, skip the length byte and compare the value
135  return persistence_->Compare(l.GetOffset() + util::getVarintLength(length_l), (void*) value_, value_size_);
136  }
137 
138  // we do not know the length, first get it, then compare
139  char buf[8];
140  persistence_->GetBuffer(l.GetOffset(), buf, 8);
141 
142  length_l = util::decodeVarint((uint8_t*)buf);
143 
144  TRACE("check equality, 3rd buffer %d %d", l.GetOffset(), value_size_);
145  return persistence_->Compare(l.GetOffset() + util::getVarintLength(length_l), (void*) value_, value_size_);
146  }
147 
148  private:
149  const char* value_;
150  size_t value_size_;
151  PersistenceT* persistence_;
152  HashCodeTypeT hashcode_;
153  };
