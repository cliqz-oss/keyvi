61 inline size_t get_common_prefix_length(const std::string& first,
62  const std::string& second) {
63  size_t common_prefix_length = 0;
64 
65  while (first[common_prefix_length] == second[common_prefix_length] &&

156 template <class PersistenceT, class ValueStoreT = internal::NullValueStore,

160  explicit Generator(const generator_param_t& params = generator_param_t(),
161  ValueStoreT* value_store = NULL)
162  : params_(params) {
163  memory_limit_ =
164  util::mapGetMemory(params_, MEMORY_LIMIT_KEY, DEFAULT_MEMORY_LIMIT);
165 
166  // use 50% or limit minus 200MB for the memory limit of the hashtable
167  const size_t memory_limit_minimization =
168  memory_limit_ > (400 * 1024 * 1024) ?
169  memory_limit_ - (200 * 1024 * 1024) :
170  memory_limit_ / 2;
171 
172  params_[TEMPORARY_PATH_KEY] = util::mapGetTemporaryPath(params);
173  minimize_ = util::mapGetBool(params_, MINIMIZATION_KEY, true);
174 
175  persistence_ = new PersistenceT(memory_limit_ - memory_limit_minimization,
176  params_[TEMPORARY_PATH_KEY]);
177 
178  stack_ = new internal::UnpackedStateStack<PersistenceT>(persistence_, 30);
179  builder_ = new internal::SparseArrayBuilder<PersistenceT, OffsetTypeT,
180  HashCodeTypeT>(
181  memory_limit_minimization, persistence_, ValueStoreT::inner_weight,
182  minimize_);
183 
184  if (value_store != NULL) {
185  value_store_ = value_store;
186  } else {
187  value_store_ = new ValueStoreT(params_);
188  }
189  }

211  void Add(const std::string& input_key,
212  typename ValueStoreT::value_t value = ValueStoreT::no_value) {
213  if (state_ != generator_state::FEEDING) {
214  throw generator_exception("not in feeding state");
215  }
216  const size_t commonPrefixLength =
217  get_common_prefix_length(last_key_, input_key);
218 
219  // keys are equal, just return
220  if (commonPrefixLength == input_key.size() &&
221  last_key_.size() == input_key.size()) {
222  return;
223  }
224 
225  // check which stack can be consumed (packed into the sparse array)
226  ConsumeStack(commonPrefixLength);
227 
228  // put everything that is not common between the two strings (the suffix)
229  // into the stack
230  FeedStack(commonPrefixLength, input_key);
231 
232  // get value and mark final state
233  bool no_minimization = false;
234  uint64_t value_idx = value_store_->GetValue(value, no_minimization);
235 
236  stack_->InsertFinalState(input_key.size(), value_idx, no_minimization);
237 
238  // count number of entries
239  ++number_of_keys_added_;
240 
241  // if inner weights are used update them
242  uint32_t weight = value_store_->GetWeightValue(value);
243  if (weight > 0) {
244  stack_->UpdateWeights(0, input_key.size() + 1, weight);
245  }
246 
247  last_key_ = input_key;
248  state_ = generator_state::FEEDING;
249  }

256  void Add(const std::string& input_key, const ValueHandle& handle) {
257  if (state_ != generator_state::FEEDING) {
258  throw generator_exception("not in feeding state");
259  }
260 
261  const size_t commonPrefixLength =

264  // keys are equal, just return

271  ConsumeStack(commonPrefixLength);
272

275  FeedStack(commonPrefixLength, input_key);

292  void CloseFeeding() {
293  if (state_ != generator_state::FEEDING) {
294  throw generator_exception("not in feeding state");
295  }
296 
297  state_ = generator_state::FINALIZING;
298 
299  // Consume all but stack[0].
300  ConsumeStack(0);
301 
302  // handling of last State.
303  internal::UnpackedState<PersistenceT>* unpackedState = stack_->Get(0);
304 
305  start_state_ = builder_->PersistState(*unpackedState);
306 
307  TRACE("wrote start state at %d", start_state_);
308  TRACE("Check first transition: %d/%d %s", (*unpackedState)[0].label,
309  persistence_->ReadTransitionLabel(start_state_ +
310  (*unpackedState)[0].label),
311  (*unpackedState)[0].label ==
312  persistence_->ReadTransitionLabel(start_state_ +
313  (*unpackedState)[0].label)
314  ? "OK"
315  : "BROKEN");
316 
317  // free structures that are not needed anymore
318  delete stack_;
319  stack_ = 0;
320  number_of_states_ = builder_->GetNumberOfStates();
321  delete builder_;
322  builder_ = 0;
323 
324  persistence_->Flush();
325 
326  state_ = generator_state::COMPILED;
327  }

404  inline void FeedStack(const size_t start, const std::string& key) {
405  for (size_t i = start; i < key.size(); ++i) {
406  uint32_t ukey = static_cast<uint32_t>(static_cast<unsigned char>(key[i]));
407  stack_->Insert(i, ukey, 0);
408  }
409 
410  // remember highest stack
411  if (key.size() > highest_stack_) {
412  highest_stack_ = key.size();
413  }
414  }
415 
416  inline void ConsumeStack(const size_t end) {
417  while (highest_stack_ > end) {
418  // Get outgoing transitions from the stack.
419  internal::UnpackedState<PersistenceT>* unpackedState =
420  stack_->Get(highest_stack_);
421 
422  OffsetTypeT transitionPointer = builder_->PersistState(*unpackedState);
423 
424  // Save transition_pointer in previous stack, indicate whether it makes
425  // sense continuing minimization
426  stack_->PushTransitionPointer(highest_stack_ - 1, transitionPointer,
427  unpackedState->GetNoMinimizationCounter());
428 
429  // Delete state
430  stack_->Erase(highest_stack_);
431 
432  --highest_stack_;
433  }
434  }
