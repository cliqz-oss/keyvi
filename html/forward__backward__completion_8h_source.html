59  MatchIterator::MatchIteratorPair GetCompletions(
60  const std::string& query, int number_of_results = 10) {
61 
62  // get query length
63  const size_t query_length = query.size();
64 
65  // get tokens
66  std::vector<std::string> strs;
67  boost::split(strs, query, boost::is_any_of("\t "));
68  size_t number_of_tokens = strs.size();
69 
70  TRACE("Query: %s length %d tokens %d", query, query_length, number_of_tokens);
71 
72  // priority queue for pruning results
73  util::BoundedPriorityQueue<uint32_t> best_scores(
74  2 * number_of_results);
75  std::vector<Match> results;
76 
77  for (auto match : forward_completions_.GetCompletions(query,
78  number_of_results)) {
79 
80  uint32_t weight = boost::lexical_cast<uint32_t>(
81  match.GetAttribute("weight"));
82 
83  // put the weight into the priority queue
84  best_scores.Put(weight);
85  match.SetScore(weight);
86  results.push_back(match);
87 
88  TRACE("Forward Completion: %s %d", match.GetMatchedString().c_str(), match.GetScore());
89  }
90 
91  if (results.size() > 0 && query_length > 4) {
92  std::make_heap(results.begin(), results.end(), result_compare());
93 
94  std::vector<Match> results_forward_and_backward;
95 
96  do {
97  std::pop_heap(results.begin(), results.end(), result_compare());
98  Match m = results.back();
99  results.pop_back();
100 
101  std::string phrase = m.GetMatchedString();
102 
103  // heuristic: stop expanding if phrase has a lower score than the worst best score
104  if (best_scores.Back() > m.GetScore()) {
105  TRACE("Stop backward completions score to low %d", m.GetScore());
106  break;
107  }
108 
109  std::reverse(phrase.begin(), phrase.end());
110  // put a space at the end to avoid infix-style backward expansion, see bugtracker #161
111  phrase.append(" ");
112 
113  TRACE("Do backward completion for %s (%d / %d)", m.GetMatchedString().c_str(), m.GetScore(), best_scores.Back());
114 
115  uint32_t last_weight = 0;
116  for (auto match : backward_completions_.GetCompletions(
117  phrase.c_str(), number_of_results)) {
118 
119  uint32_t weight = boost::lexical_cast<uint32_t>(
120  match.GetAttribute("weight"));
121 
122  if (weight < best_scores.Back()) {
123  TRACE("Skip Backward, score to low %d", weight);
124  // optimization: if score is falling again, results do not get better
125  if (last_weight > weight) {
126  TRACE("Stop Backward, no better results");
127  break;
128  }
129  continue;
130  }
131 
132  last_weight = weight;
133 
134  // accept the result
135  best_scores.Put(weight);
136  match.SetScore(weight);
137 
138  // reverse the matched string
139  std::string matched_string = match.GetMatchedString();
140  std::reverse(matched_string.begin(), matched_string.end());
141 
142  match.SetMatchedString(matched_string);
143 
144  results_forward_and_backward.push_back(match);
145 
146  TRACE("Backward Completion add: %s %d", match.GetMatchedString().c_str(), match.GetScore());
147  }
148 
149  // add the forward completion as well
150  results_forward_and_backward.push_back(m);
151 
152  } while (results.size() > 0);
153 
154  TRACE("Done backward completions");
155 
156  bool last_character_is_space = query[query_length - 1] == ' ';
157  // check for space
158  if (last_character_is_space || number_of_tokens > 1) {
159  std::make_heap(results_forward_and_backward.begin(),
160  results_forward_and_backward.end(), result_compare());
161 
162  std::string phrase = query;
163  boost::trim(phrase);
164 
165  std::reverse(phrase.begin(), phrase.end());
166  // put a space at the end to avoid infix-style backward expansion, see bugtracker #161
167  phrase.append(" ");
168  TRACE("Do backward forward completions");
169 
170  // reuse results vector
171  results.clear();
172  for (auto match : backward_completions_.GetCompletions(
173  phrase.c_str(), number_of_results)) {
174  std::string matched_string = match.GetMatchedString();
175  std::reverse(matched_string.begin(), matched_string.end());
176  // if the original query had a space at the end, this result should as well
177  if (last_character_is_space) {
178  matched_string.append(" ");
179  }
180 
181  uint32_t weight = boost::lexical_cast<uint32_t>(
182  match.GetAttribute("weight"));
183  match.SetScore(weight);
184  match.SetMatchedString(matched_string);
185 
186  results.push_back(match);
187  TRACE("Backward Completion from query add: %s %d", match.GetMatchedString().c_str(), match.GetScore());
188  }
189 
190  if (results.size() > 0) {
191  std::make_heap(results.begin(), results.end(), result_compare());
192 
193  do {
194  std::pop_heap(results.begin(), results.end(), result_compare());
195  Match m = results.back();
196  results.pop_back();
197 
198  std::string phrase = m.GetMatchedString();
199  TRACE("Do forward from backward completion for %s (%d / %d)", m.GetMatchedString().c_str(), m.GetScore(), best_scores.Back());
200 
201  // heuristic: stop expanding if phrase has a lower score than the worst best score
202  if (best_scores.Back() > m.GetScore()) {
203  TRACE("Stop backward forward completions scores to low %d", m.GetScore());
204  break;
205  }
206 
207  // match forward with this
208  for (auto match_forward : forward_completions_.GetCompletions(
209  m.GetMatchedString().c_str(), number_of_results)) {
210 
211  uint32_t weight = boost::lexical_cast<uint32_t>(
212  match_forward.GetAttribute("weight"));
213 
214  if (weight < best_scores.Back()) {
215  TRACE("Skip Backward forward, score to low %d", weight);
216  break;
217  }
218 
219  // accept the result
220  best_scores.Put(weight);
221  match_forward.SetScore(weight);
222 
223  results_forward_and_backward.push_back(match_forward);
224 
225  TRACE("Backward Forward Completion add: %s %d", match_forward.GetMatchedString().c_str(), match_forward.GetScore());
226  }
227  }while (results.size() > 0);
228  }
229  }
230 
231  std::swap(results, results_forward_and_backward);
232 
233  }

237  struct delegate_payload {
238  delegate_payload(std::vector<Match>& r)
239  : results(std::move(r)) {

242  std::vector<Match> results;
243  Match last_result;
244  };

246  std::shared_ptr<delegate_payload> data(new delegate_payload(results));

248  auto tfunc =
249  [data, number_of_results] () {
250 
251  if (data->results.size()) {
252  std::pop_heap (data->results.begin(),data->results.end(), result_compare());
253 
254  // de-duplicate
255  while (data->last_result.GetMatchedString() == data->results.back().GetMatchedString())
256  {
257  data->results.pop_back();
258  if (data->results.size() == 0){
259  return Match();
260  }
261 
262  std::pop_heap (data->results.begin(),data->results.end(), result_compare());
263  }
264 
265  data->last_result = data->results.back();
266  data->results.pop_back();
267 
268  return data->last_result;
269  }
270 
271  return Match();
272  };
