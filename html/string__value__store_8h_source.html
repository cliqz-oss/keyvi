104  template<class PersistenceT>
105  struct StringPointerForCompare
106  final
107  {
108  public:
109  StringPointerForCompare(const std::string& value,
110  PersistenceT* persistence)
111  : value_(value),
112  persistence_(persistence)
113  {
114  hashcode_ = std::hash<value_t>()(value);
115  length_ = value.size();
116  }
117 
118  int GetHashcode() const {
119  return hashcode_;
120  }
121 
122  bool operator==(const StringPointer& l) const {
123  // First filter - check if hash code is the same
124  if (l.GetHashcode() != hashcode_) {
125  return false;
126  }
127 
128  size_t length_l = l.GetLength();
129 
130  if (length_l < USHRT_MAX && length_l != length_) {
131  return false;
132  }
133 
134  size_t offset = l.GetOffset();
135 
136  // ensure not to go over memory boundaries
137  if (persistence_->size() < offset + length_ ) {
138  return false;
139  }
140 
141  for (size_t i = 0; i < length_; ++i) {
142  char c = persistence_->operator[](offset+i);
143  if (c != value_[i]) {
144  return false;
145  }
146  }
147 
148  // strings must be equal
149  return true;
150  }
151 
152  private:
153  std::string value_;
154  PersistenceT* persistence_;
155  int32_t hashcode_;
156  size_t length_;
157  };

175  uint64_t GetValue(const value_t& value, bool& no_minimization) {
176  const StringPointerForCompare<std::vector<char>> stp(value, &string_values_);
177 
178  const StringPointer p = hash_.Get(stp);
179 
180  if (!p.IsEmpty()){
181  // found the same value again, minimize
182  return p.GetOffset();
183  }
184 
185  no_minimization = true;
186 
187  // else persist string value
188  uint64_t pt = static_cast<uint64_t>(string_values_.size());
189 
190  for (size_t i = 0; i < value.size(); ++i) {
191  string_values_.push_back(value.c_str()[i]);
192  }
193 
194  // add zero termination
195  string_values_.push_back(0);
196 
197  hash_.Add(StringPointer(pt,stp.GetHashcode(),value.size()));
198 
199  return pt;
200  }
