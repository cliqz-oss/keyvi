38 class KeyViFile {
39  using ptree=boost::property_tree::ptree;
40 public:
41  explicit KeyViFile(const std::string& filename)
42  : file_stream_(filename, std::ios::binary)
43  {
44  using namespace ::boost;
45  using namespace fsa::internal;
46 
47  if (!file_stream_.good()) {
48  throw std::invalid_argument("file not found");
49  }
50 
51  char magic[KEYVI_FILE_MAGIC_LEN];
52  file_stream_.read(magic, KEYVI_FILE_MAGIC_LEN);
53  // check magic
54  if (std::strncmp(magic, KEYVI_FILE_MAGIC, KEYVI_FILE_MAGIC_LEN)){
55  throw std::invalid_argument("not a keyvi file");
56  }
57 
58  automata_properties_ = SerializationUtils::ReadJsonRecord(file_stream_);
59  persistence_offset_ = file_stream_.tellg();
60 
61  if (lexical_cast<int> (automata_properties_.get<std::string>("version")) < KEYVI_FILE_VERSION_MIN) {
62  throw std::invalid_argument("this version of keyvi file is unsupported");
63  }
64 
65  const ptree sparse_array_properties = SerializationUtils::ReadJsonRecord(file_stream_);
66 
67  if (lexical_cast<int> (sparse_array_properties.get<std::string>("version")) < KEYVI_FILE_PERSISTENCE_VERSION_MIN) {
68  throw std::invalid_argument("this versions of keyvi file is unsupported");
69  }
70 
71  const bool compact_size = lexical_cast<uint32_t> (sparse_array_properties.get<std::string>("version")) == 2;
72  const size_t bucket_size = compact_size ? sizeof(uint16_t) : sizeof(uint32_t);
73  const size_t array_size = lexical_cast<size_t>(sparse_array_properties.get<std::string>("size"));
74 
75  // check for file truncation
76  file_stream_.seekg((size_t)file_stream_.tellg() + array_size + bucket_size * array_size - 1);
77  if (file_stream_.peek() == EOF) {
78  throw std::invalid_argument("file is corrupt(truncated)");
79  }
80 
81  file_stream_.get();
82  value_store_offset_ = file_stream_.tellg();
83  }
84 
85  ptree automataProperties() const {
86  return automata_properties_;
87  }
88 
89  std::istream& persistenceStream() {
90  return file_stream_.seekg(persistence_offset_);
91  }
92 
93  std::istream& valueStoreStream() {
94  return file_stream_.seekg(value_store_offset_);
95  }
96 
97 private:
98  std::ifstream file_stream_;
99  ptree automata_properties_;
100  std::streampos persistence_offset_;
101  std::streampos value_store_offset_;
102 };
