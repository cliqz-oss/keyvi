83  template<std::size_t TsizeOther>
84  inline void SetVector(const BitVector<TsizeOther>& other, const size_t start_bit) {
85  size_t bytePosition = start_bit >> 5;
86  size_t bitPosition = start_bit & 31;
87  size_t write_length = std::min(bits_.size() - bytePosition, other.bits_.size());
88 
89  if (bitPosition == 0)
90  {
91  for (auto i = 0; i<write_length; i++){
92  bits_[bytePosition + i] |= other.bits_[i];
93  }
94  } else {
95  bits_[bytePosition] |= ((uint32_t) other.bits_[0]<< bitPosition);
96  for (auto i = 1; i<write_length; i++){
97  bits_[bytePosition + i] |= ((other.bits_[i]<< bitPosition) | (other.bits_[i-1] >> (32 - bitPosition)));
98  }
99  bits_[bytePosition + write_length] |= (other.bits_[write_length-1] >> (32 - bitPosition));
100  }
101  }

143  inline int GetNextNonSetBit(size_t start_bit) const {
144  size_t bytePosition = start_bit >> 5;
145  size_t bitPosition = start_bit & 31;
146 
147  uint32_t a = GetUnderlyingIntegerAtPosition(bytePosition, bitPosition);

164  template<std::size_t TsizeOther>
165  inline bool Disjoint(const BitVector<TsizeOther>& other, const size_t start_bit) const {
166  size_t bytePosition = start_bit >> 5;
167  size_t lenthToCheck = std::min(other.bits_.size(),
168  bits_.size() - bytePosition);
169  size_t bitPosition = start_bit & 31;
170 
171  for (size_t i = 0; i < lenthToCheck; ++i) {
172  uint32_t b = other.bits_[i];
173  if (b != 0) {
174  uint32_t a = GetUnderlyingIntegerAtPosition(bytePosition, bitPosition);
175 
176  if ((a & b) != 0) {
177  // shift until it fits
178  return false;
179  }
180  }
181 
182  ++bytePosition;
183  }
184 
185  return true;
186  }

293  inline int GetMinimumNumberOfShifts(uint32_t b, uint32_t a) const {
294  int shifts = 1;
295  a = a >> 1;
296  while ((a & b) != 0) {
297  a = a >> 1;
298  ++shifts;
299  }
300 
301  return shifts;
302  }

304  inline uint32_t GetUnderlyingIntegerAtPosition(const size_t byte_position,
305  const size_t bit_position) const {
306  if (bit_position == 0) {
307  return bits_[byte_position];
308  }
309 
310  if (byte_position + 1 < bits_.size()) {
311  return (bits_[byte_position] >> bit_position)
312  | (bits_[byte_position + 1] << (32 - bit_position));
313  }
314 
315  return bits_[byte_position] >> bit_position;
316  }
