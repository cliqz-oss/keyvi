63  bool operator<(const SegmentIterator& rhs) const {
64  // very important difference in semantics: we have to ensure that in case
65  // of equal key,

130  void Add(const std::string& filename) {
131 
132  if (std::count(inputFiles_.begin(), inputFiles_.end(), filename)) {
133  throw std::invalid_argument("File is added already: " + filename);
134  }
135 
136  fsa::automata_t fsa;
137  if (append_merge_) {
138  fsa.reset(new fsa::Automata(filename, loading_strategy_types::lazy, false));
139  } else {
140  fsa.reset(new fsa::Automata(filename));
141  }
142 
143  if (fsa->GetValueStoreType() != ValueStoreT::GetValueStoreType()) {
144  throw std::invalid_argument("Dictionaries must have the same type.");
145  }
146 
147  const auto segment_iterator = SegmentIterator(fsa::EntryIterator(fsa), segments_pqueue_.size());
148  if (!segment_iterator) {
149  return;
150  }
151 
152  segments_pqueue_.push(segment_iterator);
153  inputFiles_.push_back(filename);
154  dicts_to_merge_.push_back(fsa);
155  }

166  void Merge(const std::string& filename) {
167  using GeneratorAdapter =
168  fsa::GeneratorAdapterInterface<PersistenceT, ValueStoreT>;
169 
170  size_t sparse_array_size_sum = 0;
171  for (auto fsa: dicts_to_merge_) {
172  sparse_array_size_sum += fsa->SparseArraySize();
173  }
174 
175  ValueStoreT* value_store =
176  append_merge_ ? new ValueStoreT(inputFiles_) : new ValueStoreT(params_);
177 
178  auto generator = GeneratorAdapter::CreateGenerator(sparse_array_size_sum,
179  params_, value_store);
180 
181  std::string top_key;
182 
183  while (!segments_pqueue_.empty()) {
184  auto segment_it = segments_pqueue_.top();
185  segments_pqueue_.pop();
186 
187  top_key = segment_it.entryIterator().GetKey();
188 
189  // check for same keys and merge only the most recent one
190  while (!segments_pqueue_.empty() and segments_pqueue_.top().entryIterator().operator==(top_key)) {
191 
192  auto to_inc = segments_pqueue_.top();
193 
194  segments_pqueue_.pop();
195  if (++to_inc) {
196  TRACE("push iterator");
197  segments_pqueue_.push(to_inc);
198  }
199  }
200 
201  fsa::ValueHandle handle;
202  handle.no_minimization = false;
203 
204  // get the weight value, for now simple: does not require access to the
205  // value store itself
206  handle.weight =
207  value_store->GetMergeWeight(segment_it.entryIterator().GetValueId());
208 
209  if (append_merge_) {
210  handle.value_idx = value_store->GetMergeValueId(
211  segment_it.segmentIndex(), segment_it.entryIterator().GetValueId());
212  } else {
213  handle.value_idx = value_store->GetValue(
214  segment_it.entryIterator()
215  .GetFsa()
216  ->GetValueStore()
217  ->GetValueStorePayload(),
218  segment_it.entryIterator().GetValueId(), handle.no_minimization);
219  }
220 
221  TRACE("Add key: %s", top_key.c_str());
222  generator->Add(std::move(top_key), handle);
223 
224  if (++segment_it) {
225  segments_pqueue_.push(segment_it);
226  }
227  }
228 
229  dicts_to_merge_.clear();
230 
231  TRACE("finished iterating, do final compile.");
232 
233  generator->CloseFeeding();
234 
235  generator->SetManifestFromString(manifest_);
236  generator->WriteToFile(filename);
237  }
