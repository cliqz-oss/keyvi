56 template<class BucketT = uint16_t>
57 class SparseArrayPersistence
58 final {
59  public:
60  SparseArrayPersistence(size_t memory_limit,
61  boost::filesystem::path temporary_path)
62  : in_memory_buffer_offset_(0) {
63  buffer_size_ = memory_limit / (sizeof(unsigned char) + sizeof(BucketT));
64 
65  // align it to 16bit (for fast memcpy)
66  buffer_size_ += 16 - (buffer_size_ % 16);
67  flush_size_ = (buffer_size_ * 3) / 5;
68  // align it to 16bit
69  flush_size_ += 16 - (flush_size_ % 16);
70 
71  TRACE("Memory Limit: %d buffer size: %d flush size: %d", memory_limit, buffer_size_, flush_size_);
72 
73  labels_ = new unsigned char[buffer_size_];
74  std::memset(labels_, 0, buffer_size_);
75 
76  temporary_directory_ = temporary_path;
77  temporary_directory_ /= boost::filesystem::unique_path(
78  "dictionary-fsa-%%%%-%%%%-%%%%-%%%%");
79  boost::filesystem::create_directory(temporary_directory_);
80 
81  // size of external memory chunk: not more than 1 or 4 GB
82  size_t external_memory_chunk_size = std::min(flush_size_ * 2, (size_t)1073741824);
83 
84  // the chunk size must be a multiplier of the flush_size
85  external_memory_chunk_size = external_memory_chunk_size - (external_memory_chunk_size % flush_size_);
86 
87  labels_extern_ = new MemoryMapManager(external_memory_chunk_size,
88  temporary_directory_,
89  "characterTableFileBuffer");
90 
91  transitions_ = new BucketT[buffer_size_];
92  std::memset(transitions_, 0, buffer_size_ * sizeof(BucketT));
93 
94  transitions_extern_ = new MemoryMapManager(external_memory_chunk_size * sizeof(BucketT),
95  temporary_directory_,
96  "valueTableFileBuffer");
97  }

113  void BeginNewState(size_t offset) {
114  while ((offset + COMPACT_SIZE_WINDOW + NUMBER_OF_STATE_CODINGS)
115  >= (buffer_size_ + in_memory_buffer_offset_)) {
116  FlushBuffers();
117  }
118 
119  if (offset > highest_state_begin_) {
120  highest_state_begin_ = offset;
121  TRACE("new highest state: %d", offset);
122  }
123  }

125  void WriteTransition(size_t offset, unsigned char transitionId,
126  BucketT transitionPointer) {
127 
128  TRACE("write transition at %d with %d/%d", offset, transitionId, transitionPointer);
129 
130  highest_raw_write_bucket_ = std::max(highest_raw_write_bucket_, offset);
131 
132  if (offset >= in_memory_buffer_offset_) {
133 #ifdef PERSISTENCE_DEBUG
134  assert (labels_[offset - in_memory_buffer_offset_] == 0);
135  assert (transitions_[offset - in_memory_buffer_offset_] == 0);
136 #endif
137 
138  labels_[offset - in_memory_buffer_offset_] = transitionId;
139  transitions_[offset - in_memory_buffer_offset_] = transitionPointer;
140  return;
141  }
142 
143  unsigned char* label_ptr = (unsigned char*) labels_extern_->GetAddress(offset);
144  *label_ptr = transitionId;
145 
146  BucketT* transition_ptr = (BucketT*) transitions_extern_->GetAddress(
147  offset * sizeof(BucketT));
148  *transition_ptr = HostOrderToPesistenceOrder(transitionPointer);
149  }

179  void Flush() {
180  // make idempotent, so it can be called twice or more);
181  if (labels_) {
182  size_t highest_write_position = std::max(highest_state_begin_ + MAX_TRANSITIONS_OF_A_STATE, highest_raw_write_bucket_ + 1);
183 
184  labels_extern_->Append(labels_,
185  (highest_write_position - in_memory_buffer_offset_));
186 
187  // in place re-write
188  HostOrderToPersistenceOrder(transitions_, highest_write_position - in_memory_buffer_offset_);
189 
190  transitions_extern_->Append(transitions_,
191  (highest_write_position - in_memory_buffer_offset_) * sizeof(BucketT));
192 
193  delete[] labels_;
194  delete[] transitions_;
195  labels_ = 0;
196  transitions_ = 0;
197  }

201  boost::property_tree::ptree pt;
202  pt.put("version", GetVersion());
203

205 
206  pt.put("size",

208 
209  internal::SerializationUtils::WriteJsonRecord(stream, pt);

211 
212  labels_extern_->Write(stream,
213  highest_write_position);
214 
215  TRACE("Wrote Labels, stream at %d", stream.tellp());
216

218  stream, (highest_write_position) * sizeof(BucketT));
219  TRACE("Wrote Transitions, stream at %d", stream.tellp());
220  }

235  inline void FlushBuffers() {
236  labels_extern_->Append(labels_,
237  flush_size_);
238 
239  TRACE ("Write labels from %d to %d (flushsize %d)", in_memory_buffer_offset_, in_memory_buffer_offset_ + flush_size_, flush_size_);
240 
241  // in place re-write
242  HostOrderToPersistenceOrder(transitions_, flush_size_);
243 
244  transitions_extern_->Append(transitions_,
245  flush_size_ * sizeof(BucketT));
246 
247  size_t overlap = buffer_size_ - flush_size_;
248 
249  std::memcpy(labels_, labels_ + flush_size_, overlap);
250  std::memcpy(transitions_, transitions_ + flush_size_,
251  sizeof(BucketT) * overlap);
252 
253  std::memset(labels_ + overlap, 0, flush_size_);
254  std::memset(transitions_ + overlap, 0, sizeof(BucketT) * flush_size_);
255 
256  in_memory_buffer_offset_ += flush_size_;
257  }

292  template<>
293  inline uint64_t SparseArrayPersistence<uint16_t>::ResolveTransitionValue(size_t offset, uint16_t value) const {
294  uint16_t pt = value;
295  uint64_t resolved_ptr;
296 
297  if ((pt & 0xC000) == 0xC000) {
298  //TRACE("Compact Transition uint16 absolute");
299 
300  resolved_ptr = pt & 0x3FFF;
301  //TRACE("Compact Transition after resolve %d", resolved_ptr);
302  return resolved_ptr;
303  }
304 
305  if (pt & 0x8000){
306  // clear the first bit
307  pt &= 0x7FFF;
308  size_t overflow_bucket = (pt >> 4) + offset - 512;
309 
310  if (overflow_bucket >= in_memory_buffer_offset_) {
311  resolved_ptr = util::decodeVarshort(transitions_ -in_memory_buffer_offset_ + overflow_bucket);
312  } else {
313  if (transitions_extern_->GetAddressQuickTestOk(overflow_bucket * sizeof(uint16_t), 5)) {
314  resolved_ptr = util::decodeVarshort((uint16_t*) transitions_extern_->GetAddress(overflow_bucket * sizeof(uint16_t)));
315  } else {
316  // value might be on the chunk border, take a secure approach
317  uint16_t buffer[10];
318  transitions_extern_->GetBuffer((offset + FINAL_OFFSET_TRANSITION) * sizeof(uint16_t), buffer, 10 * sizeof(uint16_t));
319 
320  resolved_ptr = util::decodeVarshort(buffer);
321  }
322  }
323 
324  resolved_ptr = (resolved_ptr << 3) + (pt & 0x7);
325 
326  if (pt & 0x8){
327  // relative coding
328  resolved_ptr = offset - resolved_ptr + 512;
329  }
330 
331  } else {
332  resolved_ptr = offset -pt + 512;
333  }
334 
335  return resolved_ptr;
336  }

340  if (offset + FINAL_OFFSET_TRANSITION >= in_memory_buffer_offset_) {
341 
342  return util::decodeVarshort(transitions_ + offset - in_memory_buffer_offset_ + FINAL_OFFSET_TRANSITION);
343  }
344 
345  if (transitions_extern_->GetAddressQuickTestOk((offset + FINAL_OFFSET_TRANSITION) * sizeof(uint16_t), 5)) {
346  uint16_t* ptr = (uint16_t*) transitions_extern_->GetAddress((offset + FINAL_OFFSET_TRANSITION) * sizeof(uint16_t));
347  return util::decodeVarshort(ptr);
348  }
349 
350  // value might be on the chunk border, take a secure approach
351  uint16_t buffer[10];
352  transitions_extern_->GetBuffer((offset + FINAL_OFFSET_TRANSITION) * sizeof(uint16_t), buffer, 10 * sizeof(uint16_t));
353 
354  return util::decodeVarshort(buffer);
