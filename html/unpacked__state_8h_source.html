182  inline int64_t GetHashcode() {
183  if (hashcode_ == -1) {
184  int64_t b;
185  int64_t a = b = 0x9e3779b9;
186  int64_t c = weight_ > 0 ? 1 : 0;
187  int sz = used_;
188  for (int i = 0; i < sz; ++i) {
189  Transition t = outgoing_[i];
190  a += t.label;
191  b += t.value;
192  if (i < sz - 1) {
193  ++i;
194  t = outgoing_[i];
195  a += t.label << 16;
196  b += t.value << 16;
197  }
198 
199  // good old Bob Jenkins Hash
200  a -= b;
201  a -= c;
202  a ^= c >> 13;
203  b -= c;
204  b -= a;
205  b ^= a << 8;
206  c -= a;
207  c -= b;
208  c ^= b >> 13;
209  a -= b;
210  a -= c;
211  a ^= c >> 12;
212  b -= c;
213  b -= a;
214  b ^= a << 16;
215  c -= a;
216  c -= b;
217  c ^= b >> 5;
218  a -= b;
219  a -= c;
220  a ^= c >> 3;
221  b -= c;
222  b -= a;
223  b ^= a << 10;
224  c -= a;
225  c -= b;
226  c ^= b >> 15;
227  }
228 
229  hashcode_ = c;
230  }
231 
232  return hashcode_;
233  }

235  template<class OffsetTypeT, class HashCodeTypeT>
236  bool operator==(const PackedState<OffsetTypeT, HashCodeTypeT>& l) {
237 
238  // First filter - check if hash code and the number of transitions is the same
239  if (l.GetHashcode() != static_cast<HashCodeTypeT>(GetHashcode())
240  || l.GetNumberOfOutgoingTransitions() != used_) {
241  return false;
242  }
243 
244  // The number of transitions is the same. Verify that they also look the same.
245  for (int i = 0; i < used_; ++i) {
246  Transition t = outgoing_[i];
247  int label = t.label;
248 
249  if (label < FINAL_OFFSET_TRANSITION) {
250  // Is there a transition of this kind?
251  if (persistence_->ReadTransitionLabel(l.GetOffset() + label) != label) {
252  return false;
253  }
254 
255  // Does this transition lead to the same target state?
256  uint64_t target = persistence_->ReadTransitionValue(l.GetOffset() + label);
257  target = persistence_->ResolveTransitionValue(l.GetOffset() + label, target);
258  if (t.value != target) {
259  return false;
260  }
261  } else // (label == FINAL_OFFSET_TRANSITION)
262  {
263  if (persistence_->ReadTransitionLabel(l.GetOffset() + label) != FINAL_OFFSET_CODE) {
264  return false;
265  }
266 
267  // check if l has final info
268  const uint64_t value = persistence_->ReadFinalValue(l.GetOffset());
269 
270  if (t.value != value) {
271  return false;
272  }
273  }
275  }
276 
277  // note: we do not compare the weight entry because if one state has weight and the other not the hashes differ
278 
279  // all checks succeeded, states must be equal.
280  return true;
281  }
