76  OffsetTypeT PersistState(UnpackedState<SparseArrayPersistence<uint16_t>>& unpacked_state) {
77  if (unpacked_state.GetNoMinimizationCounter() == 0) {
78  // try to find a match of two equal states to minimize automata
79  const PackedState<OffsetTypeT, HashCodeTypeT> existing = state_hashtable_->Get(unpacked_state);
80  if (!existing.IsEmpty()) {
81  // if we are hitting this line minimization succeeded
82  TRACE("found minimization, equal state: %d this->weight %d", existing.GetOffset(), unpacked_state.GetWeight());
83  OffsetTypeT offset = existing.GetOffset();
84  if (unpacked_state.GetWeight() > 0) {
85  UpdateWeightIfNeeded(offset, unpacked_state.GetWeight());
86  }
87 
88  return offset;
89  }
90  } TRACE("no minimization found, write a new state");
91 
92  // minimization failed, all predecessors of this state will not be minimized, so stop trying
93  unpacked_state.IncrementNoMinimizationCounter();
94  OffsetTypeT offset = FindFreeBucket(unpacked_state);
95  //TRACE("write at %d", offset);
96 
97  WriteState(offset, unpacked_state);
98  ++number_of_states_;
99  const PackedState<OffsetTypeT, HashCodeTypeT> packed_state(offset, static_cast<HashCodeTypeT>(unpacked_state.GetHashcode()) ,
100  unpacked_state.size());
101 
102  // if minimization failed several time in a row while the minimization hash has decent amount of data,
103  // do not push the state to the minimization hash to avoid unnecessary overhead
104  if (minimize_ && (number_of_states_ < 1000000
105  || unpacked_state.GetNoMinimizationCounter() < 8)) {
106  state_hashtable_->Add(packed_state);
107  }
108 
109  return offset;
110  }
111 
112  // todo: this is not correct for compact mode!!!
113  size_t GetSize() const {
114  return (highest_persisted_state_ + MAX_TRANSITIONS_OF_A_STATE) * 5;
115  }

122  return number_of_states_;
123  }

138  OffsetTypeT FindFreeBucket(
139  UnpackedState<SparseArrayPersistence<uint16_t>>& unpacked_state) const {
140 
141  // states (state ids) start with 1 as 0 is reserved to mark a 'none-state'
142  OffsetTypeT start_position =
143  highest_persisted_state_ > SPARSE_ARRAY_SEARCH_OFFSET ?
144  highest_persisted_state_ - SPARSE_ARRAY_SEARCH_OFFSET : 1;
145 
146  // further shift it taking the first outgoing transition and find the slot where it fits in
147  start_position = taken_positions_in_sparsearray_.NextFreeSlot(
148  start_position + unpacked_state[0].label) - unpacked_state[0].label;
149 
150  do {
151  TRACE ("Find free position, probing %d", start_position);
152  start_position = state_start_positions_.NextFreeSlot(start_position);
153 
154  if (zerobyte_scrambling_state_start_positions_.IsSet(start_position)) {
155  TRACE("clash with zerobyte start position, skip it.");
156  ++start_position;
157  continue;
158  }
159 
160  if (unpacked_state.IsFinal()) {
161  if (state_start_positions_.IsSet(
162  start_position + NUMBER_OF_STATE_CODINGS)) {
163  ++start_position;
164  //TRACE ("clash with start positions, jump to next position");
165 
166  continue;
167  }
168  }
169 
170  int shift = taken_positions_in_sparsearray_.IsAvailable(
171  unpacked_state.get_BitVector(), start_position);
172 
173  if (shift == 0) {
174 
175  // check for potential conflict with existing state which could become final if the current state has
176  // a outgoing transition with label 1
177  if (start_position > NUMBER_OF_STATE_CODINGS
178  && unpacked_state.HasLabel(FINAL_OFFSET_CODE)
179  && state_start_positions_.IsSet(
180  start_position - NUMBER_OF_STATE_CODINGS)) {
181 
182  ++start_position;
183  TRACE ("interference with other state, continue search");
184 
185  continue;
186  }
187 
188  if (unpacked_state[0].label != 0 && !taken_positions_in_sparsearray_.IsSet(start_position)) {
189  TRACE("Need special handling for zero-byte state, position %ld", start_position);
190 
191  // state has no 0-byte, we have to 'scramble' the 0-byte to avoid a ghost state
192  if (start_position >= NUMBER_OF_STATE_CODINGS) {
193  OffsetTypeT zerobyte_scrambling_state = state_start_positions_.NextFreeSlot(start_position - NUMBER_OF_STATE_CODINGS);
194 
195  if (zerobyte_scrambling_state >= start_position) {
196  // unable to scramble zero byte position
197  TRACE ("unable to scramble zero byte position, continue search");
198  ++start_position;
199  continue;
200  }
201 
202  unsigned char zerobyte_scrambling_label = static_cast<unsigned char> (start_position - zerobyte_scrambling_state);
203  // avoid finalizing a state by mistake
204  if (zerobyte_scrambling_label == FINAL_OFFSET_CODE && state_start_positions_.IsSet(
205  start_position - NUMBER_OF_STATE_CODINGS)) {
206 
207  TRACE ("unable to scramble zero byte position (state finalization), continue search");
208  ++start_position;
209  continue;
210  }
211 
212  TRACE("Found zero byte label %d ,position %ld", zerobyte_scrambling_label, zerobyte_scrambling_state);
213 
214  unpacked_state.SetZerobyteState(zerobyte_scrambling_state);
215  unpacked_state.SetZerobyteLabel(zerobyte_scrambling_label);
216  }
217  }
218 
219  TRACE ("found slot at %d", start_position);
220  return start_position;
221  }
222 
223  TRACE ("state does not fit in, got shift of %d (statesize %d)", shift, unpacked_state
224  .size());
225 
226  start_position += shift;
227  } while (true);
228 
229  // not reachable
230  return -1;
231  }

233  void WriteState(const OffsetTypeT offset,
234  const UnpackedState<SparseArrayPersistence<uint16_t>>& unpacked_state) {
235  int i;
236  int len = unpacked_state.size();
237  uint32_t weight = unpacked_state.GetWeight();
238 
239  if (offset > highest_persisted_state_) {
240  highest_persisted_state_ = offset;
241  }
242 
243  persistence_->BeginNewState(offset);
244 
245  if (unpacked_state[0].label != 0) {
246  // make sure no other state is placed at offset - 256, which could cause interference
247  if ((unpacked_state[0].label == 1) && offset >= NUMBER_OF_STATE_CODINGS) {
248  state_start_positions_.Set(offset - NUMBER_OF_STATE_CODINGS);
249  }
250 
251  TRACE ("no zero byte, need special handling");
252 
253  // check if something is already written there
254  if (!taken_positions_in_sparsearray_.IsSet(offset)) {
255  if (offset >= NUMBER_OF_STATE_CODINGS) {
256  // block the position as a possible start state
257  zerobyte_scrambling_state_start_positions_.Set(unpacked_state.GetZerobyteState());
258  }
259 
260  // write the zerobyte label (it can get overridden later, which is ok)
261  WriteTransition(offset, unpacked_state.GetZerobyteLabel(), 0);
262  }
263  } else {
264  // first bit is a 0 byte, so check [1]
265  // make sure no other state is placed at offset - 256, which could cause interference
266  if (unpacked_state.size() > 1 && (unpacked_state[1].label == 1) && offset >= NUMBER_OF_STATE_CODINGS) {
267  state_start_positions_.Set(offset - NUMBER_OF_STATE_CODINGS);
268  }
269 
270  TRACE ("zero byte to be written");
271  }
272 
273  //TRACE ("WriteState at offset %d, state size %d", offset, unpacked_state.size());
274 
275  // 1st pass: reserve the buckets in the sparse array
276  taken_positions_in_sparsearray_.SetVector(unpacked_state.get_BitVector(),
277  offset);
278 
279  if (unpacked_state.IsFinal()) {
280  // Make sure no other state is placed at offset + 255, which could cause interference
281  state_start_positions_.Set(offset + NUMBER_OF_STATE_CODINGS);
282  }
283 
284 //#define STATE_WRITING_DEBUG
285 #ifdef STATE_WRITING_DEBUG
286  for (i = 0; i < len; ++i) {
287  typename UnpackedState<SparseArrayPersistence<uint16_t>>::Transition e = unpacked_state[i];
288  if (e.label < FINAL_OFFSET_TRANSITION) {
289  if (!taken_positions_in_sparsearray_.IsSet(offset + e.label)) {
290  std::cerr << "transition bit not set " << offset<< " " << e.label << std::endl;
291  std::cerr << "last transition bit: " << unpacked_state[len-1].label << std::endl;
292  std::cerr << "check bit vector: " << unpacked_state.get_BitVector().Get(e.label) << std::endl;
293 
294  }
295  } else {
296  if (e.label == FINAL_OFFSET_TRANSITION) {
297 
298  size_t vshort_size = util::getVarshortLength(e.value);
299  for (size_t i=0; i < vshort_size; ++i) {
300  if (!taken_positions_in_sparsearray_.IsSet(offset + FINAL_OFFSET_TRANSITION + i)) {
301  std::cerr << "final state bit not set " << offset<< " " << i << " " << vshort_size << " " << e.value << std::endl;
302  }
303  //taken_positions_in_sparsearray_.Set(offset + FINAL_OFFSET_TRANSITION + i);
304  }
305 
306  // Make sure no other state is placed at offset + 255, which could cause interference
307  if (!state_start_positions_.IsSet(offset + NUMBER_OF_STATE_CODINGS)) {
308  std::cerr << "state marker not set " << std::endl;
309 
310  }
311  }
312  }
313  }
314 #endif
315  // no other state should start at this offset
316  state_start_positions_.Set(offset);
317 
318  // 2nd pass: write the actual values into the buckets
319  for (i = 0; i < len; ++i) {
320  typename UnpackedState<SparseArrayPersistence<uint16_t>>::Transition e = unpacked_state[i];
321  if (e.label < FINAL_OFFSET_TRANSITION) {
322  WriteTransition(offset + e.label, e.label, e.value);
323  } else {
324  if (e.label == FINAL_OFFSET_TRANSITION) {
325  WriteFinalTransition(offset, e.value);
326  TRACE("Write final marker at %d, value %d", offset, e.value);
327  }
328  }
329  }
330 
331  if (weight) {
332  //TRACE("Write inner weight at %d, value %d", offset, weight);
333  // as all states have this, no need to code it specially
334  UpdateWeightIfNeeded(offset, weight);
335  }
336  }

338  inline void UpdateWeightIfNeeded(const size_t offset,
339  const uint32_t weight)
340  {
341  TRACE("Check for Update Weight");
342  auto n_weight =
343  (weight < COMPACT_SIZE_INNER_WEIGHT_MAX_VALUE) ?
344  weight : COMPACT_SIZE_INNER_WEIGHT_MAX_VALUE;
345 
346  if (persistence_->ReadTransitionValue(
347  offset + INNER_WEIGHT_TRANSITION_COMPACT) < n_weight) {
348  TRACE("Update weight from %d to %d", persistence_->ReadTransitionValue(offset + INNER_WEIGHT_TRANSITION_COMPACT), n_weight);
349  persistence_->WriteTransition(offset + INNER_WEIGHT_TRANSITION_COMPACT, 0,
350  n_weight);
351  // it might be, that the slot is not taken yet
352  taken_positions_in_sparsearray_.Set(
353  offset + INNER_WEIGHT_TRANSITION_COMPACT);
354 
355  // block this bucket for the start of a new state
356  state_start_positions_.Set(offset + INNER_WEIGHT_TRANSITION_COMPACT);
357  }
358  }

379  inline void WriteTransition(size_t offset, unsigned char transitionId,
380  uint64_t transitionPointer)
381  {
382  TRACE("Write offset: %ld, label: %d", offset, transitionId);
383  size_t difference = SIZE_MAX;
384 
385  if (offset + 512 > transitionPointer) {
386  difference = offset + 512 - transitionPointer;
387  }
388 
389  if (difference < COMPACT_SIZE_RELATIVE_MAX_VALUE) {
390 
391  ushort diff_as_short = static_cast<ushort>(difference);
392 
393  TRACE("Transition fits in uint16 relative: %d->%d (%d)", offset, transitionPointer, diff_as_short);
394 
395  persistence_->WriteTransition(offset, transitionId, diff_as_short);
396  return;
397  }
398 
399  if (transitionPointer < COMPACT_SIZE_ABSOLUTE_MAX_VALUE) {
400  TRACE("Transition fits in uint16 absolute: %d->%d", offset, transitionPointer);
401 
402  ushort absolute_compact_coding = static_cast<ushort>(transitionPointer)
403  | 0xC000;
404  persistence_->WriteTransition(offset, transitionId,
405  absolute_compact_coding);
406  return;
407  }
408 
409  TRACE("Transition requires overflow %d->%d", offset, transitionPointer);
410 
411  // pt to overflow bucket with various other codings
412  // set first bit to indicate overflow
413  ushort pt_to_overflow_bucket = 0x8000;
414 
415  size_t overflow_code = transitionPointer;
416 
417  if (difference < transitionPointer) {
418  // do relative coding
419  // set corresponding bit
420  pt_to_overflow_bucket |= 0x8;
421  overflow_code = difference;
422  }
423 
424  auto transitionPointer_low = overflow_code & 0x7; // get the lower part
425  auto transitionPointer_high = overflow_code >> 3; // the higher part
426 
427  // else overflow encoding
428  uint16_t vshort_pointer[8];
429  size_t vshort_size = 0;
430 
431  util::encodeVarshort(transitionPointer_high, vshort_pointer, &vshort_size);
432 
433  // find free spots in the sparse array where the pointer fits in
434  size_t start_position = offset > 512 ? offset - 512 : 0;
435  size_t zerobyte_scrambling_state = 0;
436  unsigned char zerobyte_scrambling_label = 0xff;
437 
438  for (;;) {
439  start_position = taken_positions_in_sparsearray_.NextFreeSlot(
440  start_position);
441 
442  // prevent that states without a weight get a 'zombie weight'.
443  // check that we do not write into a bucket that is used for an inner weight of another transition
444  if (inner_weight_
445  && state_start_positions_.IsSet(
446  start_position + INNER_WEIGHT_TRANSITION_COMPACT)) {
447  //TRACE("found clash wrt. weight transition, skipping %d", start_position);
448 
449  start_position += 1;
450  continue;
451  }
452 
453  if (taken_positions_in_sparsearray_.IsSet(start_position)) {
454  TRACE("Start position taken at %d",start_position);
455  start_position += 1;
456  continue;
457  }
458 
459  size_t found_slots = 1;
460 
461  for (; found_slots < vshort_size; found_slots++) {
462 
463  if (taken_positions_in_sparsearray_.IsSet(start_position + found_slots)) {
464  start_position += found_slots + 1;
465  found_slots = 0;
466  break;
467  }
468  // check that we do not write into a bucket that is used for an inner weight of another transition
469  if (inner_weight_
470  && state_start_positions_.IsSet(
471  start_position + found_slots - INNER_WEIGHT_TRANSITION_COMPACT)) {
472  TRACE("found clash wrt. weight transition, skipping %d", start_position);
473 
474  start_position += found_slots + 1;
475  found_slots = 0;
476  break;
477  }
478  }
479 
480  if (found_slots > 0 && start_position >= NUMBER_OF_STATE_CODINGS) {
481 
482  // ensure enough space: if vshort has length 2, label must start from 0xfe
483  zerobyte_scrambling_state = state_start_positions_.NextFreeSlot(start_position + vshort_size - NUMBER_OF_STATE_CODINGS - 1);
484 
485  if (zerobyte_scrambling_state >= start_position) {
486  TRACE("Did not find a state to scramble zero-bytes, no good start position, skipping %d", start_position);
487 
488  // we can probable advance more if this happens
489  start_position += found_slots + 1;
490  found_slots = 0;
491  } else {
492  zerobyte_scrambling_label = static_cast<unsigned char> (start_position - zerobyte_scrambling_state);
493 
494  if (zerobyte_scrambling_label == FINAL_OFFSET_CODE){
495  TRACE("Did not find a state to scramble zero-bytes, skipping %d", start_position);
496 
497  // we can probable advance more if this happens
498  start_position += found_slots + 1;
499  found_slots = 0;
500  }
501  }
502  }
503 
504  if (found_slots == vshort_size) {
505  break;
506  }
507  }
508 
509  TRACE("Write Overflow transition at %d, length %d", start_position, vshort_size);
510 
511  // block the pseudo state used for zerobyte scrambling
512  // state_start_positions_.Set(zerobyte_scrambling_state);
513  zerobyte_scrambling_state_start_positions_.Set(zerobyte_scrambling_state);
514  // write the overflow pointer using scrambled zerobyte labels
515  for (size_t i = 0; i < vshort_size; ++i) {
516  taken_positions_in_sparsearray_.Set(start_position + i);
517  persistence_->WriteTransition(start_position + i,
518  static_cast<unsigned char>(zerobyte_scrambling_label + i), vshort_pointer[i]);
519  }
520 
521  // encode the pointer to that bucket
522  size_t overflow_bucket = (512 + start_position) - offset;
523  pt_to_overflow_bucket |= overflow_bucket << 4;
524 
525  // add the lower part (4 bits)
526  pt_to_overflow_bucket += transitionPointer_low;
527 
528  persistence_->WriteTransition(offset, transitionId, pt_to_overflow_bucket);
529  }
