121  void operator++(int) {
122 
123  // ignore cases where we are already at the end
124  if (current_state_ == 0) {
125  TRACE("end of traversal");
126  return;
127  }
128 
129  TRACE("finding next state");
130  traversal_entry_t traversal_entry;
131  for (;;) {
132  uint64_t child_node = 0;
133  uint32_t weight = 0;
134  traversal_entry = entry_traversal_stack_.back();
135  TRACE("remaining transitions at %d %d", current_depth_, traversal_entry.size());
136 
137  if (traversal_entry.size()) {
138  // in case the priority queue got better items between creating the transition list and now,
139  // we have to skip here
140  // note: child_node remains 0, so all remaining transitions are dropped later on
141  weight = traversal_entry.front().first;
142  if (weight >= priority_queue_.Back()) {
143  current_label_ = traversal_entry.front().second;;
144  entry_traversal_stack_.back().pop_front();
145  child_node = fsa_->TryWalkTransition(current_state_, current_label_);
146  TRACE ("Walk transition with label %c (%d)", current_label_, current_label_);
147  }
148  }
149 
150  if (child_node) {
151  /* Found a valid child node
152  * go one level down
153  */
154  TRACE("found child node");
155 
156  //current_label_ = traversal_stack_[current_depth_];
157  ++current_depth_;
158  state_traversal_stack_.push_back(current_state_);
159  current_state_ = child_node;
160 
161  // todo: refactor
162  TRACE("moving down to depth: %d", current_depth_);
163  GetNextTransitionsInSortedOrder(weight);
164 
165  return;
166 
167  } else {
168  // did not found any more transitions at this deep
169 
170  if (current_depth_) {
171  /* we did not find any path at the current level (deep)
172  * go one level up
173  */
174  TRACE("current path exhausted going up");
175 
176  //traversal_stack_.pop_back();
177  current_state_ = state_traversal_stack_.back();
178  state_traversal_stack_.pop_back();
179  entry_traversal_stack_.pop_back();
180  --current_depth_;
181  } else {
182  // we are at the very end
183  TRACE("reached the end, all states traversed");
184 
185  current_state_ = 0;
186  current_depth_ = 0;
187  current_label_ = 0;
188  return;
189  }
190  }
191  }

225  void GetNextTransitionsInSortedOrder(uint32_t parent_weight) {
226  uint64_t child_node;
227  traversal_entry_t outgoing_transitions;
228  for (int i = 1; i < 256; ++i) {
229  child_node = fsa_->TryWalkTransition(current_state_, i);
230  if (child_node) {
231  // todo: stop reading the weight dependent on the depth of traversal
232  uint32_t weight = fsa_->GetWeightValue(child_node);
233 
234  // if weight is not set take the weight of the parent
235  if (weight == 0) {
236  TRACE("Received no weight, take weight from parent.");
237  weight = parent_weight;
238  }
239 
240  TRACE("transition with weight %d queue last: %d", weight, priority_queue_.Back());
241  if (weight < priority_queue_.Back()) {
242  continue;
243  } else if (parent_weight != weight && weight > priority_queue_.Back()) {
244  priority_queue_.Put(weight);
245  }
246 
247  outgoing_transitions.push_back(
248  std::pair<uint32_t, unsigned char>(weight, i));
249 
250  }
251  }
252 
253  std::sort(outgoing_transitions.begin(), outgoing_transitions.end(),
254  compare);
255  TRACE("first transition after sort: %c %d", outgoing_transitions[0].second, outgoing_transitions[0].first);
256 
257  TRACE("number of transitions found: %d", outgoing_transitions.size());
258  entry_traversal_stack_.push_back(outgoing_transitions);
259  }
