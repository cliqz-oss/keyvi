64 template<>
65 inline void TraversalState<NearTransition>::PostProcess(TraversalPayload<NearTransition>& payload) {
66  // check if we are still matching exact, if not mark it
67  if (payload.exact) {
68  TRACE("exact match at %d", payload.current_depth);
69 
70  payload.exact_depth = payload.current_depth;
71 
72  if (traversal_state_payload.position != 0) {
73  TRACE("Stop exact match at %d", payload.current_depth);
74  payload.exact = false;
75  }
76  }
77 }

80 template<>
81 inline void TraversalState<NearTransition>::Add(uint64_t s, unsigned char l, TraversalPayload<NearTransition>& payload) {
82  // check exact match
83  TRACE("Add %c depth %d, exact %c", l, payload.current_depth, payload.lookup_key[payload.current_depth]);
84 
85  if (payload.exact && payload.current_depth < payload.lookup_key.size() && payload.lookup_key[payload.current_depth] == l) {
86  // fill in and set position 0, so that we start traversal there
87  TRACE("Found exact match");
88  traversal_state_payload.position = 0;
89  traversal_state_payload.transitions[0] = NearTransition(s, l);
90  return;
91  }
92  traversal_state_payload.transitions.push_back(NearTransition(s, l));
93 }
