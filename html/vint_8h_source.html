42 template<typename int_t = uint64_t>
43 void encodeVarint(int_t value, uint8_t* output, size_t* outputSizePtr) {
44  size_t outputSize = 0;
45  //While more than 7 bits of data are left, occupy the last output byte
46  // and set the next byte flag
47  while (value > 127) {
48  //|128: Set the next byte flag
49  output[outputSize] = ((uint8_t) (value & 127)) | 128;
50 
51  //Remove the seven bits we just wrote
52  value >>= 7;
53  outputSize++;
54  }
55  output[outputSize++] = ((uint8_t) value) & 127;
56  *outputSizePtr = outputSize;
57 }
